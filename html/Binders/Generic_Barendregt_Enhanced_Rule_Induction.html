Theory Generic_Barendregt_Enhanced_Rule_Induction

theory Generic_Barendregt_Enhanced_Rule_Induction
  imports "MRBNF_Recursor"
begin

declare [[inductive_internals]]

(* General infrastructure: *)
context infinite begin

definition small :: "'a set ⇒ bool" where
"small A ≡ |A| <o |UNIV::'a set|"(* small/bounded sets *)
definition isPerm :: "('a ⇒ 'a) ⇒ bool" (* being a permutation *) where
"isPerm σ ≡ bij σ ∧ |supp σ| <o |UNIV::'a set|"

lemma small_Un: "small A ⟹ small B ⟹ small (A ∪ B)"
using Un_bound small_def infinite_UNIV by blast

lemma finite_UN_small:
assumes "finite As" and "⋀A. A ∈ As ⟹ small A"
shows "small (⋃ As)"
using assms apply(induct As)
using small_Un by (auto simp: infinite_UNIV small_def)

lemma isPerm_bij: "⋀σ. isPerm σ ⟹ bij σ"
unfolding isPerm_def by auto

lemma isPerm_id: "isPerm id" unfolding isPerm_def bij_def
  by (simp add: supp_id_bound)

lemma isPerm_comp: "isPerm σ ⟹ isPerm τ ⟹ isPerm (σ o τ)"
by (meson infinite_UNIV bij_comp isPerm_def supp_comp_bound)

lemma isPerm_inv: "⋀σ. isPerm σ ⟹ isPerm (inv σ)"
by (simp add: isPerm_def supp_inv_bound)

lemma small_isPerm:
assumes s: "small A" "small B" "small A'" "A ∩ A' = {}"
shows "∃σ. isPerm σ ∧ σ ` A ∩ B = {} ∧ (∀a∈A'. σ a = a)"
proof-
  obtain D where D: "D ∩ B = {}" "D ∩ A = {}" "D ∩ A' = {}" and DA: "|D| =o |A|"
  using exists_subset_compl[of _ A "A' ∪ B"]
  by (metis Field_card_of Int_Un_emptyI1 Int_Un_emptyI2 Int_commute card_of_Card_order card_of_UNIV
   cinfinite_def infinite_UNIV ordIso_symmetric s(1-3) small_Un small_def)

  then obtain u where u: "bij_betw u A D"
  using card_of_ordIso ordIso_symmetric by blast
  hence u: "u ` A = D" "⋀a b. a ∈ A ⟹ b ∈ A ⟹ u a = u b ⟷ a = b"
  unfolding bij_betw_def inj_on_def by auto

  let ?iu = "inv_into A u"
  have iu: "bij_betw ?iu D A"
  using u by (metis bij_betw_def bij_betw_inv_into inj_on_def)
  hence iu: "?iu ` D = A" "⋀a b. a ∈ D ⟹ b ∈ D ⟹ ?iu a = ?iu b ⟷ a = b"
  unfolding bij_betw_def inj_on_def by auto

  define v where "v ≡ λa. if a ∈ A then u a else if a ∈ D then ?iu a else a"
  have v[simp]: "⋀a. a ∈ A ⟹ v a = u a" "⋀a. a ∈ D ⟹ v a = ?iu a"
  "⋀a. a ∉ A ∧ a ∉ D ⟹ v a = a"
  using D(2) unfolding v_def by auto

  have cas: "⋀a. a ∈ A ∨ a ∈ D ∨ (a ∉ A ∧ a ∉ D)" by auto

  have bv: "bij v"
  unfolding bij_def inj_def image_def apply auto
  apply (metis (mono_tags, lifting) D(2) Int_emptyD f_inv_into_f imageI iu(1) u(1) u(2) v_def)
  by (metis f_inv_into_f inv_into_into iu(1) u(1) v(2) v_def)

  have sv: "supp v ⊆ A ∪ D" unfolding supp_def using v(3) by blast

  show ?thesis apply(rule exI[of _ v], intro conjI)
    subgoal using bv sv s(1) unfolding isPerm_def small_def
      by (meson DA card_of_Un_ordLess_infinite card_of_subset_bound infinite_UNIV ordIso_ordLess_trans)
    subgoal using D(1) u(1) by auto
    subgoal using sv D(3) s(4) unfolding supp_def by auto .
qed

lemma isPerm_invL: "isPerm σ ⟹ σ o inv σ = id"
by (meson bij_is_surj isPerm_bij surj_iff)

lemma isPerm_invL': "isPerm σ ⟹ σ (inv σ a) = a"
using isPerm_invL pointfree_idE by fastforce

lemma isPerm_invR: "isPerm σ ⟹ inv σ o σ = id"
by (meson bij_def inv_o_cancel isPerm_bij)

lemma isPerm_invR': "isPerm σ ⟹ inv σ (σ a) = a"
using isPerm_invR pointfree_idE by fastforce

lemma small_empty[simp,intro!]: "small {}"
  by (simp add: infinite_UNIV small_def)

lemma small_singl[simp,intro!]: "small {x}"
  by (simp add: infinite_UNIV small_def)

lemma small_two[simp,intro!]: "small {x,y}"
  by (simp add: infinite_UNIV small_def)

lemma small_three[simp,intro!]: "small {x,y,z}"
  by (simp add: infinite_UNIV small_def)

lemma small_image: "small B ⟹ small (σ ` B)"
using card_of_image ordLeq_ordLess_trans small_def by blast

lemmas image_comp' = image_comp[symmetric]

end (* context infinite *)


locale CComponents =
  (* 'T: term-like entities, ranged over by t *)
fixes Tperm :: "('A :: infinite ⇒ 'A) ⇒ 'T ⇒ 'T"
and Tsupp :: "'T ⇒ 'A set"
  (* 'B: binder-like entities, ranged over by xs *)
and Bperm :: "('A ⇒ 'A) ⇒ 'B ⇒ 'B"
and Bsupp :: "'B ⇒ 'A set"
  (* well-formed binders: *)
and bnd :: "'B ⇒ bool"
(* smallness w.r.t. crossing binders: *)
and bsmall :: "'A set ⇒ bool"
assumes
TTperm_id[simp]: "Tperm id = id"
and
TTperm_comp: "⋀σ τ. isPerm σ ⟹ isPerm τ ⟹ Tperm (σ o τ) = Tperm σ o Tperm τ"
and
ssmall_Tsupp: "⋀t. small (Tsupp t)"
and (* the weaker, inclusion-based version is sufficient (and also for B): *)
TTsupp_seminat: "⋀t σ. isPerm σ ⟹ Tsupp (Tperm σ t) ⊆ σ ` (Tsupp t)"
and
TTsupp_supporting: "⋀t σ. isPerm σ ⟹ (∀a∈Tsupp t. σ a = a) ⟹ Tperm σ t = t"
and
BBperm_id[simp]: "Bperm id = id"
and
BBperm_comp: "⋀σ τ. isPerm σ ⟹ isPerm τ ⟹ Bperm (σ o τ) = Bperm σ o Bperm τ"
and
small_Bsupp: "⋀xs. bnd xs ⟹ small (Bsupp xs)"
and
BBsupp_seminat: "⋀xs σ. isPerm σ ⟹ Bsupp (Bperm σ xs) ⊆ σ ` (Bsupp xs)"
and
BBsupp_supporting: "⋀xs σ. isPerm σ ⟹ (∀a∈Bsupp xs. σ a = a) ⟹ Bperm σ xs = xs"
and
(* bsmallness is subject to properties similar to the ones enjoyed by smallness: *)
bsmall_Bsupp: "⋀xs. bnd xs ⟹ bsmall (Bsupp xs)"
and
bsmall_Un: "bsmall A ⟹ bsmall B ⟹ bsmall (A ∪ B)"
begin

lemma TTperm_comp': "isPerm σ ⟹ isPerm τ ⟹ Tperm (σ o τ) t = Tperm σ (Tperm τ t)"
using TTperm_comp by fastforce

lemma image_Tsupp_disj: "isPerm σ ⟹ B ∩ Tsupp t = {} ⟹ image σ B ∩ Tsupp (Tperm σ t) = {}"
using TTsupp_seminat isPerm_bij by (fastforce simp: bij_implies_inject)

lemma BBperm_comp': "isPerm σ ⟹ isPerm τ ⟹ Bperm (σ o τ) xs = Bperm σ (Bperm τ xs)"
using BBperm_comp by fastforce

lemma image_Bsupp_disj: "isPerm σ ⟹ B ∩ Bsupp xs = {} ⟹ image σ B ∩ Bsupp (Bperm σ xs) = {}"
using BBsupp_seminat isPerm_bij by (fastforce simp: bij_implies_inject)

(* *)

(* The notion of preserving (well-formed) binders: *)
definition presBnd :: "('A ⇒ 'A) ⇒ bool"
where "presBnd σ ≡ ∀xs. bnd xs ⟷ bnd (Bperm σ xs)"

lemma presBnd_id[simp,intro]: "presBnd id"
unfolding presBnd_def by auto

lemma presBnd_comp[simp]:
"isPerm σ ⟹ presBnd σ ⟹ isPerm τ ⟹ presBnd τ ⟹ presBnd (τ o σ)"
by (simp add: BBperm_comp' presBnd_def)

lemma presBnd_inv[simp]: "isPerm σ ⟹ presBnd σ ⟹ presBnd (inv σ)"
by (metis BBperm_comp' BBperm_id id_apply isPerm_inv isPerm_invL presBnd_def)

end (* locale CComponents *)



(* GENERAL VERSIONS OF THE LOCALES, WITH EXPLICIT BINDERS *)

locale IInduct1 = CComponents Tperm Tsupp Bperm Bsupp bnd bsmall
for Tperm :: "('A :: infinite ⇒ 'A) ⇒ 'T ⇒ 'T"
and Tsupp :: "'T ⇒ 'A set"
and Bperm :: "('A ⇒ 'A) ⇒ 'B ⇒ 'B"
and Bsupp :: "'B ⇒ 'A set"
and bnd and bsmall
+
fixes (* The operator that defines the inductive predicate as lfp:  *)
GG :: "'B ⇒ ('T ⇒ bool) ⇒ 'T ⇒ bool"
(* *)
assumes
GG_mmono: "⋀R R' xs t. R ≤ R' ⟹ GG xs R t ⟹ GG xs R' t"
and
GG_eequiv: "⋀σ R xs t. isPerm σ ⟹ presBnd σ ⟹
   GG xs R t ⟹ GG  (Bperm σ xs) (λt'. R (Tperm (inv σ) t')) (Tperm σ t)"
and
GG_bnd: "⋀R xs t. GG xs R t ⟹ bnd xs"
and
extend_to_presBnd:
"⋀xs A A'. bnd xs ⟹ small A ⟹ bsmall A ⟹ A' ⊆ A ⟹ Bsupp xs ∩ A' = {} ⟹
           ∃ρ. isPerm ρ ∧ presBnd ρ ∧ ρ ` Bsupp xs ∩ A = {} ∧ id_on A' ρ"
begin

lemma GG_mmono2[mono]: "⋀R R' xs t.  R ≤ R' ⟹ GG xs R t ⟶ GG xs R' t"
  using GG_mmono by blast


inductive II :: "'T ⇒ bool" where
GG_II_intro: "GG xs II t ⟹ II t"

lemma "II ≡ lfp (λR t. ∃xs. GG xs R t)"
using II_def[simplified] .

lemma II_equiv:
assumes "II t" and σ: "isPerm σ" "presBnd σ"
shows "II (Tperm σ t)"
using assms(1) proof induct
  case (GG_II_intro xs t)
  have GG: "GG xs (λt. II (Tperm σ t)) t"
  apply(rule GG_mmono[OF _ GG_II_intro(1)]) using σ by auto
  have GG: "GG (Bperm σ xs) (λt. II (Tperm σ (Tperm (inv σ) t))) (Tperm σ t)"
  using GG_eequiv[OF σ GG] .
  have "GG (Bperm σ xs) II (Tperm σ t)"
  apply(rule GG_mmono[OF _ GG])
  using σ
  by auto (metis TTperm_comp' TTperm_id id_apply isPerm_inv isPerm_invL)
  thus ?case by (subst II.simps, auto)
qed

lemma GG_mmono'[mono]: "⋀R R' xs t.  R ≤ R' ⟹
 Bsupp xs ∩ Tsupp t = {} ∧ GG xs R t ⟶ Bsupp xs ∩ Tsupp t = {} ∧ GG xs R' t"
  using GG_mmono by blast

inductive II' :: "'T ⇒ bool" where
GG_II'_intro: "Bsupp xs ∩ Tsupp t = {} ⟹ GG xs II' t ⟹ II' t"

lemma II'_imp_II: "II' t ⟹ II t"
apply(induct rule: II'.induct)
by (smt (verit) GG_mmono II.simps predicate1I)

lemma II'_equiv:
assumes "II' t" and σ: "isPerm σ" "presBnd σ"
shows "II' (Tperm σ t)"
using assms(1) proof induct
  case (GG_II'_intro xs t)  note B = GG_II'_intro(1)
  have GG: "GG xs (λt. II' (Tperm σ t)) t"
  apply(rule GG_mmono[OF _ GG_II'_intro(2)]) using σ by auto
  have BB: "Bsupp (Bperm σ xs) ∩ Tsupp (Tperm σ t) = {}" using image_Tsupp_disj[OF σ(1) GG_II'_intro(1)]
  using BBsupp_seminat[OF σ(1)] by fastforce
  have GG: "GG (Bperm σ xs) (λt. II' (Tperm σ (Tperm (inv σ) t))) (Tperm σ t)"
  using GG_eequiv[OF σ GG] .
  have 0: "(λt. II' (Tperm σ (Tperm (inv σ) t))) = II'"
  unfolding fun_eq_iff
  by (metis TTperm_comp' TTsupp_supporting σ(1) id_apply isPerm_comp isPerm_inv isPerm_invL)
  have GG: "GG  (Bperm σ xs) II'(Tperm σ t)"
  using GG unfolding 0 .
  show ?case using BB GG small_image by (subst II'.simps, auto)
qed

end (* context IInduct1 *)


locale IInduct = IInduct1 Tperm Tsupp Bperm Bsupp bnd bsmall GG
for Tperm :: "('A :: infinite ⇒ 'A) ⇒ 'T ⇒ 'T"
and Tsupp :: "'T ⇒ 'A set"
and Bperm :: "('A ⇒ 'A) ⇒ 'B ⇒ 'B"
and Bsupp :: "'B ⇒ 'A set"
and bnd and bsmall
and GG :: "'B ⇒ ('T ⇒ bool) ⇒ 'T ⇒ bool"
+
assumes
II_bsmall: "⋀t. II t ⟹ bsmall (Tsupp t)"
and
GG_rrefresh:
"⋀R xs t. (∀t. R t ⟶ II t) ⟹
         (∀σ t. isPerm σ ∧ presBnd σ ∧ R t ⟶ R (Tperm σ t)) ⟹ GG xs R t ⟹
         ∃ys. Bsupp ys ∩ Tsupp t = {} ∧ GG ys R t"



context IInduct
begin

(* NB: The following could replace GG_refresh in the axiomatization.
IIt has the advantage that it is weaker, but also two disadvantages:
-- it depends on the "auxiliary" defined predicate II'
-- the dependency on II' seems truly inessential, in that in concrete cases
all that one needs to use is the equivariance of II'
Later note: in the end, if was useful for getting in inductive information,
namely strenthening GG_refresh to further assume that R implies II ("∀t. R t ⟶ II t").
 *)
lemma GG_refresh_II':
"⋀xs t. GG xs II' t ⟹ ∃ys. Bsupp ys ∩ Tsupp t = {} ∧ GG ys II' t"
using GG_rrefresh II'_equiv by (simp add: II'_imp_II)

lemma II_imp_II': "II t ⟹ II' t"
proof(induct rule: II.induct)
  case (GG_II_intro xs t)
  hence GG: "GG xs II' t" by (metis (no_types, lifting) GG_mmono predicate1I)
  from GG_refresh_II'[OF GG]
  obtain ys where 0: "Bsupp ys ∩ Tsupp t = {}" "GG ys II' t" by auto
  show ?case using II'.intros[OF 0] .
qed

lemma II_eq_II': "II = II'"
apply(rule ext)
subgoal for t
apply(rule iffI)
  subgoal using II_imp_II' by auto
  subgoal using II'_imp_II by auto . .

end (* context IInduct *)


(* The locale with the more restrictive assumptions, in the style of Urban-Berghofer-Norrish: *)
locale IInduct_simple = IInduct1 Tperm Tsupp Bperm Bsupp bnd bsmall GG
for Tperm :: "('A :: infinite ⇒ 'A) ⇒ 'T ⇒ 'T"
and Tsupp :: "'T ⇒ 'A set"
and Bperm :: "('A ⇒ 'A) ⇒ 'B ⇒ 'B"
and Bsupp :: "'B ⇒ 'A set"
and bnd and bsmall
and GG :: "'B ⇒ ('T ⇒ bool) ⇒ 'T ⇒ bool"
+
assumes
II_bsmall: "⋀t. II t ⟹ bsmall (Tsupp t)"
and
GG_ffresh: "⋀R xs t. GG xs R t ⟹ Bsupp xs ∩ Tsupp t = {}"


sublocale IInduct_simple < IInduct apply standard
  using II_bsmall GG_ffresh by blast+


context IInduct
begin

(* Strong (Barendregt-enhanced) rule induction *)

lemma extend:
assumes xs: "bnd xs" and t: "II t" and p: "small (Psupp p)" "bsmall (Psupp p)"
and b: "Bsupp xs ∩ Tsupp t = {}"
shows "∃ρ. isPerm ρ ∧ presBnd ρ ∧ ρ ` (Bsupp xs) ∩ (Psupp p ∪ Tsupp t) = {} ∧
           id_on (Tsupp t) ρ"
apply(rule extend_to_presBnd)
  subgoal by fact
  subgoal using p(1) ssmall_Tsupp small_Un by auto
  subgoal by (simp add: II_bsmall bsmall_Un p(2) t)
  subgoal by simp
  subgoal by fact .

theorem BE_iinduct[consumes 2]:
(* Parameters: *)
fixes Psupp :: "'P ⇒ 'A set"
assumes small_Psupp: "⋀p. small (Psupp p) ∧ bsmall (Psupp p)"
(* *)
assumes II: "II (t::'T)"
and strong: "⋀ p xs t. Bsupp xs ∩ Psupp p = {} ⟹ Bsupp xs ∩ Tsupp t = {} ⟹
      GG xs (λt'. II t' ∧ (∀p'. R p' t')) t ⟹ R p t"
shows "R p t"
proof-
  {fix σ assume σ: "isPerm σ" "presBnd σ"
   have "R p (Tperm σ t)"
   using II σ unfolding II_eq_II' proof(induct arbitrary: σ p)
     fix xs t σ p
     assume vt: "Bsupp xs ∩ Tsupp t = {}" (* this additional vt assumption is what we have gained
     by transitioning from II to II', whose inductive definition has this freshness side-condition *)
     and GG: "GG xs (λt'. II' t' ∧ (∀σ'. isPerm σ' ⟶ presBnd σ' ⟶ (∀p'. R p' (Tperm σ' t')))) t"
     and σ: "isPerm σ" "presBnd σ"

     have sp: "small (Psupp p)" "bsmall (Psupp p)" using small_Psupp by auto

     have "II' t"
       by (metis (no_types, lifting) GG II.simps II_eq_II' IInduct1.GG_mmono IInduct1_axioms predicate1I)
     hence II_T: "II t" using II'_imp_II by auto
     hence II_s_t: "II (Tperm σ t)" by (simp add: II_equiv σ(1) σ(2))

     define xs' where xs': "xs' ≡ Bperm σ xs"


     have "bnd xs" using GG_bnd[OF GG] .
     hence bnd': "bnd xs'"
     unfolding xs' using σ(2) presBnd_def by auto

     have v't: "Bsupp xs' ∩ Tsupp (Tperm σ t) = {}"
     using vt unfolding xs'
     using image_Tsupp_disj σ
     by (meson BBsupp_seminat Int_subset_empty1)

     have small_p_t: "small (Psupp p ∪ Tsupp (Tperm σ t))"
       by (simp add: small_Psupp ssmall_Tsupp small_Un)

     obtain ρ where ρ: "isPerm ρ" "presBnd ρ" "ρ ` (Bsupp xs') ∩ (Psupp p ∪ Tsupp (Tperm σ t)) = {}"
     "∀a ∈ Tsupp (Tperm σ t). ρ a = a"
     using extend[OF bnd' II_s_t, of Psupp, OF sp v't]
     unfolding id_on_def by metis

     have "ρ ` (Bsupp xs') ∩ Psupp p = {}"
     and "ρ ` (Bsupp xs') ∩ Tsupp (Tperm σ t) = {}"
     using ρ(1,2,3) by auto
     hence fresh_p: "Bsupp (Bperm ρ xs') ∩ Psupp p = {}"
     and fresh_t: "Bsupp (Bperm ρ xs') ∩ Tsupp (Tperm σ t) = {}"
     using ρ(1) BBsupp_seminat by fastforce+

     hence "Tperm ρ (Tperm σ t) = Tperm σ t"
     using TTsupp_supporting[OF ρ(1,4)] by blast
     hence 0: "Tperm (ρ o σ) t = Tperm σ t"
   	 by (simp add: TTperm_comp' ρ(1) σ)

     have ρσ: "isPerm (ρ o σ)" "presBnd (ρ o σ)" apply (simp add: ρ(1) σ isPerm_comp)
     apply(rule presBnd_comp) using σ ρ by auto

     define σ'' where σ'': "σ'' = ρ o σ"
     have ss_σ'': "isPerm σ''" "presBnd σ''" using ρ(1) σ σ'' isPerm_comp isPerm_inv apply blast
     using ρσ(2) σ'' by blast

     have 1[simp]: "σ'' ∘ inv (ρ o σ) = id"
     unfolding σ'' using ρσ isPerm_invL by auto

     have "GG xs (λt'. II' t' ∧ (∀p'. R p' (Tperm σ'' t'))) t"
     apply(rule GG_mmono[OF _ GG]) using ss_σ'' by auto
     hence GG: "GG xs (λt'. II' (Tperm σ'' t') ∧ (∀p'. R p' (Tperm σ'' t'))) t"
     using II'_equiv[OF _ ss_σ'']
     by (smt (verit, del_insts) GG_mmono predicate1I)
     have GG: "GG (Bperm (ρ o σ) xs)
                 (λt'. II' (Tperm σ'' (Tperm (inv (ρ o σ)) t')) ∧
                      (∀p'. R p' (Tperm σ'' (Tperm (inv (ρ o σ)) t'))))
                 (Tperm (ρ o σ) t) "
     using GG_eequiv[OF ρσ GG] .

     have GG: "GG (Bperm ρ xs') (λt'. II' (Tperm (σ'' o inv (ρ o σ)) t') ∧ (∀p'. R p' (Tperm (σ'' o inv (ρ o σ)) t')))
                (Tperm σ t) "
     unfolding xs' unfolding image_comp 0[symmetric] BBperm_comp'[symmetric, OF ρ(1) σ(1)]
     apply(rule GG_mmono[OF _ GG])
     by auto (metis 1 TTperm_comp' TTperm_id σ'' id_apply ss_σ''(1) isPerm_inv)+

     have GG: "GG (Bperm ρ xs') (λt'. II' t' ∧ (∀p'. R p' t')) (Tperm σ t)"
     apply(rule GG_mmono[OF _ GG])
     by auto

     show "R p (Tperm σ t)"
     using strong[OF fresh_p fresh_t GG[unfolded II_eq_II'[symmetric]]] .
  qed
  }
  from this[of id] show ?thesis
  by (simp add: isPerm_id)
qed

end (* context IInduct *)

and Tsupp :: "'T ⇒ 'A set"
assumes
Tperm_id: "Tperm id = id"
and
Tperm_comp: "⋀σ τ. isPerm σ ⟹ isPerm τ ⟹ Tperm (σ o τ) = Tperm σ o Tperm τ"

small_Tsupp: "⋀t. small (Tsupp t)"
and
Tsupp_seminat: "⋀t σ. isPerm σ ⟹ Tsupp (Tperm σ t) ⊆ σ ` (Tsupp t)"

locale Induct1 = LSNominalSet Tperm Tsupp
for Tperm :: "('A :: infinite ⇒ 'A) ⇒ 'T ⇒ 'T"
and Tsupp :: "'T ⇒ 'A set"
+
fixes (* The operator that defines the inductive predicate as lfp:  *)
G :: "'A set ⇒ ('T ⇒ bool) ⇒ 'T ⇒ bool"
assumes
G_mono: "⋀R R' B t. R ≤ R' ⟹ small B ⟹ G B R t ⟹ G B R' t"
and
G_equiv: "⋀σ R B t. isPerm σ ⟹ small B ⟹ G B R t ⟹ G (image σ B) (λt'. R (Tperm (inv σ) t')) (Tperm σ t)"
begin

definition GG where "GG B R t ≡ small B ∧ G B R t"

lemma GGG_mmono: "R ≤ R' ⟹ GG B R t ⟹ GG B R' t"
by (simp add: GG_def G_mono)

lemma GGG_eequiv: "isPerm σ ⟹
   GG B R t ⟹ GG  (σ ` B) (λt'. R (Tperm (inv σ) t')) (Tperm σ t)"
using GG_def G_equiv small_image by force

lemma GGG_bnd: "GG B R t ⟹ small B"
unfolding GG_def by auto

(* In this particular contex, all small bijections are well-formed: *)
lemma isPerm_presBnd: "isPerm σ ⟹ presBnd σ"
unfolding presBnd_def isPerm_def
using bij_card_of_ordIso ordIso_ordLess_trans ordIso_symmetric small_def by blast

lemma cinfinite_A: "cinfinite |UNIV::'A set|"
unfolding cinfinite_def
by (simp add: infinite_UNIV)

lemma extend_small:
assumes "small (A :: 'A set)" "bij_betw ρ A B" "id_on (A∩B) ρ"
shows "∃ρ'. isPerm ρ' ∧ eq_on A ρ' ρ"
using assms cinfinite_A ex_bij_betw_supp'[of "|UNIV::'A set|" A ρ B]
unfolding eq_on_def small_def isPerm_def id_on_def eq_on_def by auto

lemma extend_presBnd:
"small A ⟹ bij_betw ρ A B ⟹ id_on (A∩B) ρ ⟹ presBnd ρ ⟹
     ∃ρ'. isPerm ρ' ∧ presBnd ρ' ∧ eq_on A ρ' ρ"
using extend_small by (metis isPerm_presBnd)

lemma eextend_to_presBnd:
assumes "small B" "small A" "A' ⊆ A" "B ∩ A' = {}"
shows "∃ρ. isPerm ρ ∧ presBnd ρ ∧ ρ ` B ∩ A = {} ∧ id_on A' ρ"
proof-
  have "|- (A ∪ B)| =o |UNIV::'A set|"
  by (metis Compl_eq_Diff_UNIV Un_bound assms(1) assms(2) card_of_Un_diff_infinite infinite_UNIV small_def)
  hence "|B| <o |- (A ∪ B)|"
    using assms(1) ordIso_symmetric ordLess_ordIso_trans small_def by blast
  then obtain f where f: "inj_on f B" "f ` B ⊆ - (A ∪ B)"
    by (meson card_of_ordLeq ordLeq_iff_ordLess_or_ordIso)
  define g where "g ≡ λa. if a ∈ B then f a else a"
  have g: "inj_on g (B ∪ A')" "g ` (B ∪ A') ⊆ - (A ∪ B) ∪ A'" using f
  unfolding g_def inj_on_def using assms(3,4) by auto
  define C where C: "C = g ` (B ∪ A')"
  have b: "bij_betw g (B ∪ A') C" unfolding C bij_betw_def using g by simp

  have 0: "Cinfinite |UNIV::'A set|" "|B ∪ A'| <o |UNIV::'A set|" "eq_on ((B ∪ A') ∩ C) g id"
    subgoal by (simp add: cinfinite_A)
    subgoal by (meson assms(1-3) card_of_subset_bound small_Un small_def)
    subgoal using assms(3) f unfolding eq_on_def C g_def by auto .

  show ?thesis using ex_bij_betw_supp'[OF 0(1,2) b 0(3)] apply safe
  subgoal for ρ apply(rule exI[of _ ρ]) using isPerm_presBnd unfolding isPerm_def
  unfolding id_on_def apply auto
  apply (metis ComplD UnCI eq_on_def f(2) g_def image_subset_iff)
  by (metis Int_emptyD UnCI assms(4) eq_on_def g_def) .
qed

locale PreNominalSet =

assumes
Tperm_id[simp]: "Tperm id = id"
and

begin

"supports X t ≡ ∀σ. isPerm σ ∧ (∀x∈X. σ x = x) ⟶ Tperm σ t = t"

"isPerm σ ⟹ isPerm τ ⟹ Tperm (σ o τ) t = Tperm σ (Tperm τ t)"
using Tperm_comp by auto

assumes "supports X t"
and "isPerm σ" "∀x∈X. σ x = x"
shows "Tperm σ t = t"
using assms unfolding supports_def by auto

lemma supports2:
assumes "supports X t"
and "isPerm σ" "isPerm τ" "∀x∈X. σ x = τ x"
shows "Tperm σ t = Tperm τ t"
proof-
  have "Tperm (inv τ o σ) t = t"
  apply(rule supports)
  using assms by (auto simp add: isPerm_comp isPerm_inv isPerm_invR')
  thus ?thesis
  by (metis Tperm_comp' Tperm_id assms(2,3) id_apply isPerm_inv isPerm_invL)

(* The crucial lemma about nominal sets, all the good properties follow fro it.
The proof was a bit more technical than usually presented in the literature since
we work directly with finite permutations rather than swapping.
*)
lemma supports_int:
assumes fX: "finite X" and X: "supports X t" and fY: "finite Y" and Y: "supports Y t"
shows "supports (X ∩ Y) t"
unfolding supports_def proof safe
  fix σ assume s: "isPerm σ" and i: "∀x∈X ∩ Y. σ x = x"
  have "infinite (- (σ ` (X ∪ Y) ∪ X ∪ Y))"
    by (metis assms fX finite_UnI finite_compl finite_imageI infinite_UNIV)
  then obtain V where "V ⊆ (- (σ ` (X ∪ Y) ∪ X ∪ Y))" and cV: "card V = card (σ ` (X - Y))"
  and fV: "finite V"
  by (meson infinite_arbitrarily_large)
  hence V: "V ∩ (σ ` (X ∪ Y) ∪ X ∪ Y) = {}" by auto
  obtain f where bf: "bij_betw f (σ ` (X - Y)) V" using fV cV
    by (metis bij_betw_iff_card fX finite_Diff finite_imageI)
  have bif: "bij_betw (inv_into (σ ` (X - Y)) f) V (σ ` (X - Y))"
    using bf bij_betw_inv_into by blast
  define τ where "τ ≡ λx. if x ∈ (σ ` (X - Y)) then f x else if x ∈ V then inv_into (σ ` (X - Y)) f x else x"
  have "supp τ ⊆ σ ` (X - Y) ∪ V" unfolding supp_def τ_def by auto
  hence "finite (supp τ)"
    by (simp add: fV fX finite_subset)
  hence sst: "|supp τ| <o |UNIV::'A set|"
    by (simp add: infinite_UNIV)
  have tau: "isPerm τ" "bij_betw τ (σ ` (X - Y)) V" "id_on (σ ` Y) τ"
    subgoal unfolding isPerm_def using s V bf bif apply safe
      subgoal by (smt (verit, best) Int_Un_emptyI1 Un_Diff_cancel2 τ_def
      bij_betw_apply bij_betw_inv_into_left disjoint_iff image_Un involuntory_imp_bij)
      subgoal using sst . .
    subgoal unfolding τ_def bij_def inj_def inj_on_def bij_betw_def
    by simp (metis bf bij_betw_imp_surj_on bij_betw_inv_into_left imageI)
    subgoal unfolding id_on_def τ_def using V
    by (auto simp add: isPerm_bij s bij_implies_inject) .
  define σ' where "σ' ≡ τ ∘ σ"
  have s': "isPerm σ'" by (simp add: σ'_def isPerm_comp s tau(1))
  have bbs': "bij_betw σ' X (σ' ` X)" "bij_betw (inv σ') (σ' ` X) X" "X ∩ (σ' ` (X-Y)) = {}"
  apply (simp add: bij_imp_bij_betw isPerm_bij s')
  apply (meson bij_betw_inv_into_subset isPerm_bij s' subset_UNIV)
  by (metis Int_assoc Un_Int_eq(3) Un_Int_eq(4) V σ'_def bij_betw_imp_surj_on boolean_algebra.conj_disj_distrib image_comp inf_compl_bot_right tau(2))

  define τ' where "τ' ≡ λu. if u ∈ (X-Y) then σ' u else if u ∈ σ' ` (X-Y) then inv σ' u else u"
  have τ': "⋀u. u ∈ X ⟹ τ' u = σ' u" "⋀u. u ∈ σ' ` (X-Y) ⟹ τ' u = inv σ' u"
  "⋀u. u ∈ - (σ' ` (X-Y) ∪ (X-Y)) ⟹ τ' u = u"
  unfolding τ'_def
  apply (metis Diff_iff Int_iff σ'_def τ_def bbs'(3) bij_betw_def comp_apply empty_iff i image_comp inj_image_mem_iff isPerm_bij s tau(2))
  using bbs'(3) apply auto[1]
  by auto
  have τ'Y: "∀x∈Y. τ' x = x" unfolding τ'_def
  by (smt (verit, ccfv_threshold) IntI Int_emptyD Un_Diff_Int Un_Diff_cancel Un_iff
      V σ'_def bij_betw_imp_surj_on comp_apply f_inv_into_f i id_onD id_on_def image_comp
      image_eqI in_mono inf_sup_absorb inv_into_into isPerm_bij o_inv_distrib s s' sup_commute
      sup_ge2 tau(1) tau(2) tau(3))
  have "supp τ' ⊆ (X-Y) ∪ σ' ` (X - Y) ∪ V" unfolding supp_def τ'_def by auto
  hence "finite (supp τ')" by (simp add: fV fX finite_subset)
  hence sst': "|supp τ'| <o |UNIV::'A set|"
    by (simp add: infinite_UNIV)
  have pτ': "isPerm τ'" using s' τ' unfolding isPerm_def apply safe
    subgoal by (metis τ'_def bij_betw_inv_into image_in_bij_eq inv_simp1 involuntory_imp_bij)
    subgoal using sst' . .
  have 00: "Tperm σ' t = Tperm τ' t" apply(rule supports2[OF X])
    subgoal by fact
    subgoal by fact
    subgoal using τ' by auto .
  also have "Tperm τ' t = t" apply(rule supports[OF Y])
    subgoal by fact
    subgoal by fact .
  finally have 11: "Tperm σ' t = t" .
  have σ': "σ' ` (X-Y) ∩ Y = {}" unfolding σ'_def
  by auto (metis DiffI Int_Un_emptyI2 Int_emptyD V bij_betw_apply imageI tau(2))
  have "Tperm σ' t = Tperm σ t" apply(rule supports2[OF Y])
    subgoal by (simp add: σ'_def isPerm_comp s tau(1))
    subgoal by fact
    subgoal unfolding σ'_def by auto (meson id_onD imageI tau(3)) .
  thus "Tperm σ t = t" using 11 by auto
qed

lemma supports_Tsupp: "supports (Tsupp t) t"
proof-
  obtain Y where Y: "finite Y ∧ supports Y t"
    using ex_fin_supports by auto
  have 0: "Tsupp t = ⋂ {X ∩ Y | X . finite X ∧ supports X t}"
  unfolding Tsupp_def using Y by auto
  have 1: "{X ∩ Y |X. finite X ∧ supports X t} ⊆ Pow Y" by auto
  show ?thesis unfolding 0 apply(rule supports_finite_Int)
    subgoal using Y using "1" finite_subset by auto
    subgoal using Y using supports_int by auto .
qed
