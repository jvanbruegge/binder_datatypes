lemma exists_subset_compl:
  assumes "Cinfinite r" "r ≤o |UNIV::'b set|" "|U ∪ S::'b set| <o r"
  shows "∃B. U ∩ B = {} ∧ B ∩ S = {} ∧ |U| =o |B|"
proof -
  have 1: "|U| <o r" using assms(3) using card_of_Un1 ordLeq_ordLess_trans by blast
  have "|-(U ∪ S)| =o |UNIV::'b set|"
    using card_of_Un_diff_infinite[OF
        cinfinite_imp_infinite[OF cinfinite_mono[OF assms(2) conjunct1[OF assms(1)]]]
        ordLess_ordLeq_trans[OF assms(3,2)]
    ]
    by (simp add: Compl_eq_Diff_UNIV)
  then have "|U| <o |-(U ∪ S)|" using ordLess_ordLeq_trans[OF 1 assms(2)] ordIso_symmetric ordLess_ordIso_trans by fast
  then obtain B where 1: "B ⊆ -(U ∪ S)" "|U| =o |B|"
    by (meson internalize_card_of_ordLeq2 ordLess_imp_ordLeq)
  then have "U ∩ B = {}" "B ∩ S = {}" by blast+
  then show ?thesis using 1 by blast
qed

lemma exists_suitable_aux:
  assumes "Cinfinite r" "r ≤o |UNIV::'a set|" "P ⟹ |U ∪ (S - U)::'a set| <o r"
  shows "∃(u::'a ⇒ 'a). P ⟶ bij u ∧ |supp u| <o r ∧ imsupp u ∩ (S - U) = {} ∧ u ` U ∩ S = {}"
proof -
  have 1: "P ⟹ |U| <o r" using assms(3) using card_of_Un1 ordLeq_ordLess_trans by blast
  obtain B where 2: "P ⟹ U ∩ B = {}" "P ⟹ B ∩ (S - U) = {}" "P ⟹ |U| =o |B|"
    using exists_subset_compl[OF assms(1,2,3)] by blast
  obtain u where 3: "bij u" "P ⟹ |supp u| <o r" "P ⟹ bij_betw u U B" "P ⟹ imsupp u ∩ (S - U) = {}"
    using ordIso_ex_bij_betw_supp[OF assms(1) 1 2(3,1) Diff_disjoint 2(2)] by blast
  then have "P ⟹ u ` U ⊆ B" unfolding bij_betw_def by blast
  then have "P ⟹ u ` U ∩ S = {}" using 2 by blast
  then show ?thesis using 3 by blast
qed

definition eq_bij_betw where
  "eq_bij_betw r u w g A B x y f1 f2 h L R ≡
    bij u ∧ |supp u| <o r ∧ imsupp u ∩ g (A x) = {} ∧ u ` f1 (A x) ∩ f1 (A x) = {}
  ∧ bij w ∧ |supp w| <o r ∧ imsupp w ∩ g (B y) = {} ∧ w ` f1 (B y) ∩ f1 (B y) = {}
  ∧ eq_on (f2 y) (u ∘ L ∘ h) (w ∘ R)"

lemma exists_bij_betw:
  fixes L R h::"'a ⇒ 'a"
  assumes "Cinfinite r" "r ≤o |UNIV::'a set|" "bij R" "bij L" "bij h" "f2 x = h ` f2 y"
    and u: "|f1 (A x) ∪ g (A x)::'a set| <o r" "f1 (A x) ∩ g (A x) = {}" "f1 (A x) = L ` f2 x"
    and w: "|(f1 (B y)) ∪ (g (B y))::'a set| <o r" "f1 (B y) ∩ g (B y) = {}" "f1 (B y) = R ` f2 y"
  shows "∃(u::'a ⇒ 'a) (w::'a ⇒ 'a).
    bij u ∧ |supp u| <o r ∧ imsupp u ∩ g (A x) = {} ∧ u ` f1 (A x) ∩ f1 (A x) = {}
  ∧ bij w ∧ |supp w| <o r ∧ imsupp w ∩ g (B y) = {} ∧ w ` f1 (B y) ∩ f1 (B y) = {}
  ∧ eq_on (f2 y) (u ∘ L ∘ h) (w ∘ R)"
proof -
  have 1: "|f1 (A x)| <o r" "|f1 (B y)| <o r"
    using card_of_Un1 card_of_Un2 ordLeq_ordLess_trans u(1) w(1) by blast+
  then have 2: "|f1 (A x)| <o |UNIV::'a set|" "|f1 (B y)| <o |UNIV::'a set|"
    using ordLess_ordLeq_trans[OF _ assms(2)] by blast+
  have "|f1 (A x) ∪ g (A x) ∪ f1 (B y) ∪ g (B y)| <o r" (is "|?A| <o _")
    using Un_Cinfinite_ordLess[OF u(1) w(1) assms(1)] Un_assoc by metis
  then have "|-?A| =o |UNIV::'a set|"
    apply -
    apply (rule card_of_Un_diff_infinite[of "UNIV::'a set" ?A, unfolded Compl_eq_Diff_UNIV[symmetric]])
     apply (rule cinfinite_imp_infinite)
     apply (rule cinfinite_mono)
      apply (rule assms(2))
     apply (rule conjunct1[OF assms(1)])
    apply (rule ordLess_ordLeq_trans)
     apply assumption
    apply (rule assms(2))
    done
  then have "|f1 (A x)| <o |-?A|" by (rule ordLess_ordIso_trans[OF 2(1) ordIso_symmetric])

  then obtain C where C: "C ⊆ -?A" "|f1 (A x)| =o |C|"
    using ordLess_imp_ordLeq[THEN iffD1[OF internalize_card_of_ordLeq2]] by metis
  then have 3: "f1 (A x) ∩ C = {}" "C ∩ g (A x) = {}" "f1 (B y) ∩ C = {}" "C ∩ g (B y) = {}" by blast+

  obtain u::"'a ⇒ 'a" where x: "bij u" "|supp u| <o r" "bij_betw u (f1 (A x)) C" "imsupp u ∩ g (A x) = {}"
    using ordIso_ex_bij_betw_supp[OF assms(1) 1(1) C(2) 3(1) u(2) 3(2)] by blast

  have "bij_betw (inv R) (f1 (B y)) (f2 y)" unfolding bij_betw_def by (simp add: assms(3) inj_on_def w(3) bij_implies_inject)
  moreover have "bij_betw h (f2 y) (f2 x)" using bij_imp_bij_betw assms(5,6) by auto
  moreover have "bij_betw L (f2 x) (f1 (A x))" unfolding bij_betw_def by (simp add: assms(4) inj_on_def u(3) bij_implies_inject)
  ultimately have 4: "bij_betw (u ∘ L ∘ h ∘ inv R) (f1 (B y)) C" using bij_betw_trans x(3) by meson

  obtain w::"'a ⇒ 'a" where y: "bij w" "|supp w| <o r" "bij_betw w (f1 (B y)) C"
    "imsupp w ∩ g (B y) = {}" "eq_on (f1 (B y)) w (u ∘ L ∘ h ∘ inv R)"
    using ex_bij_betw_supp[OF assms(1) 1(2) 4 3(3) w(2) 3(4)] by blast

  have "eq_on (f2 y) (u ∘ L ∘ h) (w ∘ R)" using y(5) unfolding eq_on_def using assms(3) w(3) by auto
  moreover have "u ` f1 (A x) ∩ f1 (A x) = {}" "w ` f1 (B y) ∩ f1 (B y) = {}" using bij_betw_imp_surj_on x(3) y(3) 3(1,3) by blast+
  ultimately show ?thesis using x(1,2,4) y(1,2,4) by blast
qed

definition compSS :: "('a ⇒ 'a) ⇒ ('a ⇒ 'a) ⇒ ('a ⇒ 'a)" where
  "compSS f g ≡ f ∘ g ∘ inv f"

lemma compSS_comp0:
  fixes f g h::"'a ⇒ 'a"
  assumes "infinite (UNIV::'a set)" "bij f" "|supp f| <o |UNIV::'a set|" "bij g" "|supp g| <o |UNIV::'a set|" "|supp h| <o |UNIV::'a set|"
  shows "(compSS f ∘ compSS g) h = compSS (f ∘ g) h"
proof -
  have "|supp (g ∘ h ∘ inv g)| <o |UNIV::'a set|"
    by (simp add: supp_comp_bound assms supp_inv_bound)
  then show ?thesis unfolding compSS_def
    by (simp add: comp_assoc[symmetric] o_inv_distrib assms)
qed

lemma compSS_cong_id:
  fixes f::"'a ⇒ 'a"
  assumes "bij f" "|supp p| <o |UNIV::'a set|" and cong: "⋀a. a ∈ imsupp p ⟹ f a = a"
  shows "compSS f p = p"
proof -
  have 1: "imsupp f ∩ imsupp p = {}"
    by (meson Int_emptyI assms(1) bij_imsupp_supp_ne cong not_in_supp_alt)
  then show ?thesis unfolding compSS_def using imsupp_commute
    by (metis assms(1) bij_is_surj inv_inv_eq o_inv_o_cancel surj_imp_inj_inv)
qed

  assumes "|B| <o |UNIV::'a set|" "|B| <o |UNIV::'a set| ⟹ |A ∪ B| <o |UNIV::'a set|" "infinite (UNIV::'a set)"
  shows "∃ρ. bij ρ ∧ |supp ρ| <o |UNIV::'a set| ∧ ρ ` B ∩ A = {} ∧ ρ ` B ∩ B = {} ∧ id_on (A - B) ρ ∧ (∀x. ρ (ρ x) = x)"
  using eextend_fresh[of B "A ∪ B" "A - B", OF assms(1,2,3), OF assms(1)]
  unfolding Int_Un_distrib fun_eq_iff o_apply id_apply
  by blast
