Theory Swapping

theory Swapping
  imports "Prelim.Prelim"
begin

definition swap :: "'a ⇒ 'a ⇒ ('a ⇒ 'a)" (infix "(↔)" 60) where
  "swap a b ≡ λx. if x = a then b else if x = b then a else x"

lemma bij_swap[simp]: "bij (a ↔ b)"
  by (simp add: involuntory_imp_bij swap_def)

lemma supp_swap: "supp (a ↔ b) ⊆ {a, b}"
  by (metis insert_iff not_in_supp_alt subsetI swap_def)

lemma supp_swap_bound[simp]: "infinite (UNIV::'a set) ⟹ |supp (a ↔ b)| <o |UNIV::'a set|"
  using card_of_subset_bound[OF supp_swap] by (metis finite.emptyI finite_insert finite_ordLess_infinite2)

lemma swap_simps[simp]: "(a ↔ b) a = b" "(a ↔ b) b = a"
  "x ≠ a ⟹ x ≠ b ⟹ (a ↔ b) x = x"
  unfolding swap_def by auto

lemma swap_sym: "(a ↔ b) = (b ↔ a)"
  unfolding swap_def by force

lemma swap_inv[simp]: "inv (a ↔ b) = (a ↔ b)"
  by (smt (verit, ccfv_threshold) inv_equality swap_simps(1,2,3))
lemma swap_same[simp]: "(a ↔ a) x = x"
  by (simp add: swap_def)
lemma swap_id[simp]: "(a ↔ a) = id"
  by fastforce
lemma swap_twice[simp]: "(X ↔ Y) ((Y ↔ X) x) = x"
  unfolding swap_def by presburger
lemma swap_twice'[simp]: "(X ↔ Y) ((X ↔ Y) x) = x"
  unfolding swap_def by presburger
lemma swap_o_twice[simp]: "(X ↔ Y) ∘ (Y ↔ X) = id"
  unfolding comp_def swap_twice by (simp add: id_def)
lemma swap_o_twice'[simp]: "(X ↔ Y) ∘ (X ↔ Y) = id"
  unfolding comp_def swap_twice by (simp add: id_def)

lemma swap_apply_fresh_bij: "bij f ⟹ (z ≠ x ⟹ f z = z) ⟹ z ≠ y ⟹ (f x ↔ y) (f z) = (x ↔ y) z"
  by (simp add: bij_implies_inject swap_def)

lemma swap_apply_fresh_bij2: "bij f ⟹ (z ≠ x ⟹ f z = z) ⟹ (x ↔ f x) z = f z"
  by (metis inv_simp1 swap_def)

lemma id_on_image_Diff: "bij f ⟹ id_on (A - {x}) f ⟹ f x ∈ A ⟹ f x = x"
  by (metis (full_types) DiffI id_onD inv_simp1 singletonD)
lemma id_on_swap: "y ∉ A ∨ x = y ⟹ id_on (A - {x}) (x ↔ y)"
  by (metis Diff_iff id_onI singletonI swap_def)

end
