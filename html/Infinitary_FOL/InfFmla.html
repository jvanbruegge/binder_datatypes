definition card_k :: "k rel" where
  "card_k ≡ map_prod Abs_k Abs_k ` ( |UNIV::k1 set| +c |UNIV::k2 set| )"

lemma Field_k: "Field card_k = (UNIV :: k set)"
proof -
  let ?r = "|UNIV::k1 set| +c |UNIV::k2 set|"
  let ?ar = "λx. Abs_k (Rep_k x)"
  have 1: "⋀P. (∀x∈Field ?r. P x) = (∀x. P (Rep_k x))" using Rep_k_induct Rep_k by blast
  have 2: "⋀P. (∃x∈Field ?r. P x) = (∃x. P (Rep_k x))" using Rep_k_cases Rep_k by blast
  have 3: "⋀A a b. (a, b) ∈ A ⟹ (Abs_k a, Abs_k b) ∈ map_prod Abs_k Abs_k ` A" unfolding map_prod_def by auto
  have "∀x∈Field ?r. (∃b∈Field ?r. (x, b) ∈ ?r) ∨ (∃a∈Field ?r. (a, x) ∈ ?r)"
    unfolding Field_def Range.simps Domain.simps Un_iff by blast
  then have "∀x. (∃b. (Rep_k x, Rep_k b) ∈ ?r) ∨ (∃a. (Rep_k a, Rep_k x) ∈ ?r)" unfolding 1 2 .
  then have "∀x. (∃b. (?ar x, ?ar b) ∈ map_prod Abs_k Abs_k ` ?r) ∨ (∃a. (?ar a, ?ar x) ∈ map_prod Abs_k Abs_k ` ?r)" using 3 by fast
  then have "∀x. (∃b. (x, b) ∈ card_k) ∨ (∃a. (a, x) ∈ card_k)" unfolding card_k_def Rep_k_inverse .
  then show ?thesis unfolding Field_def Domain.simps Range.simps set_eq_iff Un_iff eqTrueI[OF UNIV_I] ex_simps simp_thms .
qed

lemma card_k_alt: "card_k = dir_image ( |UNIV::k1 set| +c |UNIV::k2 set| ) Abs_k"
  unfolding card_k_def dir_image_def by auto

lemma card_k_ordIso: " |UNIV::k1 set| +c |UNIV::k2 set| =o card_k"
unfolding card_k_alt
apply (rule dir_image_ordIso)
apply (simp add: csum_def)
apply (simp add: Abs_k_inject inj_on_def)
done

lemma fmlaP_rrename_simps[simp]:
fixes f::"'a::var_fmlaP_pre ⇒ 'a"
shows  "bij f ⟹ |supp f| <o |UNIV::'a set| ⟹ rrename_fmlaP f (Eq x1 x2) = Eq (f x1) (f x2)"
  "bij f ⟹ |supp f| <o |UNIV::'a set| ⟹ rrename_fmlaP f (Neg x) = Neg (rrename_fmlaP f x)"
  "bij f ⟹ |supp f| <o |UNIV::'a set| ⟹ rrename_fmlaP f (Conj F) = Conj (map_set⇩k⇩1 (rrename_fmlaP f) F)"
  "bij f ⟹ |supp f| <o |UNIV::'a set| ⟹ rrename_fmlaP f (All x3 x4) = All (map_set⇩k⇩2 f x3) (rrename_fmlaP f x4)"
  apply (auto simp: fmlaP_vvsubst_rrename[symmetric])[3]
  apply (unfold All_def fmlaP.rrename_cctors map_fmlaP_pre_def comp_def Abs_fmlaP_pre_inverse[OF UNIV_I]
    map_sum.simps map_prod_simp
    )
  apply (rule refl)
  done

  unfolding Bot_def fmlaP_rrename_simps map_set⇩k⇩1_def map_fun_def comp_def Abs_set⇩k⇩1_inverse[OF UNIV_I]
  unfolding id_def map_bset_bempty

lemma bij_betw_snth:
assumes V: "|V::var set| <o |UNIV::var set|"
shows "∃f vs'. bij_betw f (set⇩k⇩2 vs) (set⇩k⇩2 vs') ∧ V ∩ set⇩k⇩2 vs' = {} ∧ map_set⇩k⇩2 f vs = vs'"
proof-
  have "|UNIV - V| =o |UNIV::var set|" apply(rule card_of_Un_diff_infinite)
  using V by (auto simp: infinite_UNIV)
  hence "|set⇩k⇩2 vs| <o |UNIV - V|"
  by (metis Field_card_of ordIso_symmetric ordLeq_iff_ordLess_or_ordIso ordLess_ordIso_trans ordLess_transitive set⇩k⇩2.set_bd var_set⇩k⇩2_class.large)
  then obtain f where f: "inj_on f (set⇩k⇩2 vs)" "f ` (set⇩k⇩2 vs) ⊆ UNIV - V"
  by (meson card_of_ordLeq ordLess_imp_ordLeq)
  show ?thesis apply(intro exI[of _ f] exI[of _ "map_set⇩k⇩2 f vs"])
  using f unfolding bij_betw_def using set⇩k⇩2.set_map by fastforce
qed

lemma refresh:
  assumes V: "set⇩k⇩2 xs ∩ V = {} " "|V| <o |UNIV::var set|"
shows "∃f xs'. bij (f::var⇒var) ∧ |supp f| <o |UNIV::var set| ∧
               set⇩k⇩2 xs' ∩ set⇩k⇩2 xs = {} ∧ set⇩k⇩2 xs' ∩ V = {} ∧
               map_set⇩k⇩2 f xs = xs' ∧ id_on V f"
proof-
  have ss: "|set⇩k⇩2 xs| <o |UNIV::var set|"
  apply (rule ordLess_ordLeq_trans)
  apply (rule set⇩k⇩2.set_bd)
  using var_set⇩k⇩2_class.large by auto
  hence ss1: "|set⇩k⇩2 xs ∪ V| <o |UNIV::var set|"
  by (meson assms(2) var_fmlaP_pre_class.Un_bound)
  obtain f xs' where f: "bij_betw f (set⇩k⇩2 xs) (set⇩k⇩2 xs')"
  "set⇩k⇩2 xs ∩ set⇩k⇩2 xs' = {}" "V ∩ set⇩k⇩2 xs' = {}" "map_set⇩k⇩2 f xs = xs'"
  using bij_betw_snth[OF ss1, of xs] by fastforce
  hence iif: "inj_on f (set⇩k⇩2 xs)" unfolding bij_betw_def by auto

  obtain u where u: "bij u ∧
      |supp u| <o |UNIV::var set| ∧ bij_betw u (set⇩k⇩2 xs) (set⇩k⇩2 xs') ∧
      imsupp u ∩ V = {} ∧
      eq_on (set⇩k⇩2 xs) u f"
  using ex_bij_betw_supp_UNIV[OF _ ss f(1,2), of V] V(1) f(3)
  using infinite_UNIV by blast
  hence iiu: "inj_on u (set⇩k⇩2 xs)" unfolding bij_betw_def by auto

  show ?thesis apply(intro exI[of _ u] exI[of _ xs'])
  using u f iif iiu unfolding eq_on_def id_on_def imsupp_def supp_def
  by (smt (verit) UnCI disjoint_iff mem_Collect_eq set⇩k⇩2.map_cong)
qed
