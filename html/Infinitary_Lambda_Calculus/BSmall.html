Theory BSmall

(* Bound smallness (reusable for at least two relations... *)

theory BSmall 
imports Supervariables
begin

type_synonym B = "ivar dstream option"

fun Bperm :: "(ivar ⇒ ivar) ⇒ B ⇒ B" where 
"Bperm f xxs = (case xxs of None ⇒ None
                          |Some xs ⇒ Some (dsmap f xs))"

fun Bsupp :: "B ⇒ ivar set" where 
"Bsupp xxs = (case xxs of None ⇒ {}
                         |Some xs ⇒ dsset xs)"

fun bnd :: "B ⇒ bool" where 
"bnd xxs = (case xxs of None ⇒ True
                       |Some xs ⇒ super xs)"

definition bsmall :: "ivar set ⇒ bool" where 
"bsmall X ≡ finite (touchedSuper X)"

lemma super_Un_ddset_triv: "{xs. super xs ∧ (A ∪ B) ∩ dsset xs ≠ {}} ⊆  
   {xs. super xs ∧ A ∩ dsset xs ≠ {}} ∪ 
   {xs. super xs ∧ B ∩ dsset xs ≠ {}}"
by auto

lemma bsmall_emp[simp,intro!]: "bsmall {}" 
unfolding bsmall_def by auto


lemma bsmall_supp_ifd[simp,intro!]: "bsmall (supp id)"
unfolding bsmall_def supp_def by auto

lemma super_bsmall_dsset: "super xs ⟹ bsmall (dsset xs)"
by (simp add: bsmall_def)

lemma bsmall_supp_comp: "bsmall (supp g) ⟹ bsmall (supp f) ⟹ bsmall (supp (g o f))"
unfolding bsmall_def using touchedSuper_supp_comp 
by (metis finite_UnI rev_finite_subset)

lemma bsmall_supp_inv: 
"bij f ⟹ |supp f| <o |UNIV::ivar set| ⟹ presSuper f ⟹ bsmall (supp f) 
 ⟹ bsmall (supp (inv f))"
unfolding bsmall_def by (simp add: touchedSuper_supp_inv)


end
