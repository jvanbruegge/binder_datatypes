Theory ILC_UBeta_depth

(* Uniform (parallel) reduction that also tracks the applicative depth 
(i.e., the number of application operators on top of) of the affected redex (⇒d from Mazza) *)
theory ILC_UBeta_depth
imports ILC_Head_Reduction
begin


inductive ustepD :: "nat ⇒ ilterm stream ⇒ ilterm stream ⇒ bool" where
  Beta: "uniformS es ⟹ stream_all2 hred es es' ⟹ ustepD 0 es es'"
| iApL: "uniformS (sflat ess) ⟹ ustepD d es es' ⟹ ustepD (Suc d) (smap2 iAp es ess) (smap2 iAp es' ess)"
| iApR: "uniformS es ⟹ ustepD d (sflat ess) (sflat ess') ⟹ ustepD (Suc d) (smap2 iAp es ess) (smap2 iAp es ess')"
| Xi: "super xs ⟹ ustepD d es es' ⟹ ustepD d (smap (iLm xs) es) (smap (iLm xs) es')"

thm ustepD_def

lemma ustepD_uniformS:
assumes "ustepD d es es'" 
shows "uniformS es ∧ uniformS es'"
using assms proof(induct rule: ustepD.induct)
  case Beta 
  then show ?case using hred_uniformS by simp
next
  case iApL
  then show ?case unfolding uniformS_smap2_iAp_iff by auto
next
  case iApR 
  then show ?case unfolding uniformS_smap2_iAp_iff by auto
next
  case Xi
  then show ?case using uniformS_smap2_iLm_iff by auto
qed

lemma hred_def2: 
assumes t: "uniform t"
shows 
"hred t tt ⟷ 
   (∃xs e1 es2. super xs ∧ t = iAp (iLm xs e1) es2 ∧ tt = itvsubst (imkSubst xs es2) e1)"
unfolding hred_def by (metis iLm_switch_to_super t)

lemma hred_reneqv': 
assumes r: "reneqv e ee" "hred e e'"
shows "∃ee'. hred ee ee' ∧ reneqv e' ee'"
proof-
  have u: "uniform e" using r unfolding uniform_def by auto
  show ?thesis using r
  unfolding hred_def2[OF u] 
  apply(auto dest!: reneqv_iAp_casesL reneqv_iLm_casesL) 
  by (metis hred_def r(1) reneqv_head_reduction)
qed

lemma hred_reneqvS': 
assumes r: "reneqvS es ees" "stream_all2 hred es es'"
shows "∃ees'. stream_all2 hred ees ees' ∧ reneqvS es' ees'"
proof-
  have "∀i. ∃ee'. hred (snth ees i) ee' ∧ reneqv (snth es' i) ee'"
  using assms unfolding stream_all2_iff_snth reneqvS_def sset_range image_def
  using hred_reneqv' by simp blast
  then obtain EE' where EE': "∀i. hred (snth ees i) (EE' i) ∧ reneqv (snth es' i) (EE' i)"
  by metis
  show ?thesis apply(rule exI[of _ "smap EE' nats"])
  using EE' assms unfolding stream_all2_iff_snth reneqvS_def sset_range image_def 
  by simp (metis (no_types, lifting) hred_reneqv)
qed

definition G :: "B ⇒ (T ⇒ bool) ⇒ T ⇒ bool"
where
"G ≡ λxxs R t.  
         (∃es es'. xxs = None ∧ fst t = 0 ∧ fst (snd t) = es ∧ snd (snd t) = es' ∧ 
                   uniformS es ∧ stream_all2 hred es es')
         ∨
         (∃d es es' ess. xxs = None ∧ fst t = Suc d ∧ fst (snd t) = smap2 iAp es ess ∧ snd (snd t) = smap2 iAp es' ess ∧ 
                       uniformS (sflat ess) ∧ R (d, es, es')) 
         ∨
         (∃es d ess ess'. xxs = None ∧ fst t = Suc d ∧ fst (snd t) = smap2 iAp es ess ∧ snd (snd t) = smap2 iAp es ess' ∧ 
                        uniformS es ∧ R (d, sflat ess, sflat ess'))
         ∨
         (∃xs d es es'. xxs = Some xs ∧ fst t = d ∧ fst (snd t) = smap (iLm xs) es ∧ snd (snd t) = smap (iLm xs) es' ∧ 
                      super xs ∧ R (d, es, es'))"

lemma eextend_to_presBnd: 
assumes "bnd xxs" "small A" "bsmall A" "A' ⊆ A" "Bsupp xxs ∩ A' = {}"
shows "∃ρ. isPerm ρ ∧ presBnd ρ ∧ ρ ` Bsupp xxs ∩ A = {} ∧ id_on A' ρ" 
proof(cases xxs)
  case None
  thus ?thesis apply(intro exI[of _ id]) unfolding isPerm_def by auto
next
  case (Some xs)
  hence 0: "super xs" "|A| <o |UNIV::ivar set|" "finite (touchedSuper A)" "A' ⊆ A"
  "dsset xs ∩ A' = {}"
  using assms by (auto split: option.splits simp: small_def bsmall_def) 
  show ?thesis using extend_super[OF 0] apply safe
  subgoal for ρ apply(rule exI[of _ ρ]) 
  using Some by (auto split: option.splits simp: presBnd_presSuper isPerm_def) .
qed

(* FROM ABSTRACT BACK TO CONCRETE: *)
thm ustepD.induct[no_vars] 

corollary strong_induct_ustepD[consumes 2, case_names Beta iApL iApR Xi]: 
assumes par: "⋀p. small (Psupp p) ∧ bsmall (Psupp p)"
and st: "ustepD d t1 t2"  
and Beta: "⋀d es es' p. 
  stream_all2 hred es es' ⟹ 
  R p d es es'"
and iApL: "⋀d es es' ess p. 
  ustepD d es es' ⟹ (∀p'. R p' d es es') ⟹ 
  R p (Suc d) (smap2 iAp es ess) (smap2 iAp es' ess)"
and iApR: "⋀d ess ess' es p. 
  ustepD d (sflat ess) (sflat ess') ⟹ (∀p'. R p' d (sflat ess) (sflat ess')) ⟹ 
  R p (Suc d) (smap2 iAp es ess) (smap2 iAp es ess')"
and Xi: "⋀d es es' xs p. 
  dsset xs ∩ Psupp p = {} ⟹ 
  ustepD d es es' ⟹ (∀p'. R p' d es es') ⟹ 
  R p d (smap (iLm xs) es) (smap (iLm xs) es')" 
shows "R p d t1 t2"
unfolding ustepD_I
apply(subgoal_tac "case (d,t1,t2) of (d, t1, t2) ⇒ R p d t1 t2")
  subgoal by simp
  subgoal using par st apply(elim Ustep.strong_iinduct[where R = "λp (d,t1,t2). R p d t1 t2"])
    subgoal unfolding ustepD_I by simp
    subgoal for p B t apply(subst (asm) G_def) 
    unfolding ustepD_I[symmetric] apply(elim disjE exE)
      subgoal using Beta by auto
      subgoal using iApL by auto  
      subgoal using iApR by auto  
      subgoal using Xi by auto . . .

(* Also inferring equivariance from the general infrastructure: *)
corollary irrename_ustepD:
assumes f: "bij f" "|supp f| <o |UNIV::ivar set|" "presSuper f"
and r: "ustepD d es es'" 
shows "ustepD d (smap (irrename f) es) (smap (irrename f) es')"
using assms unfolding ustepD_I using Ustep.II_equiv[of "(d,es,es')" f]
unfolding Tperm_def isPerm_def presBnd_presSuper by auto
