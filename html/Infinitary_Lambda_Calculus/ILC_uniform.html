Theory ILC_uniform

(* Here we instantiate the general enhanced rule induction to the "reneqv" predicate from Mazza  *)
theory ILC_uniform
imports ILC_Renaming_Equivalence 
begin


definition uniform :: "ivar iterm ⇒ bool" 
where "uniform e ≡ ∃e'. reneqv e e'" 

lemma uniform_finite_touchedUponT: "uniform e ⟹ finite (touchedSuperT e)"
using reneqv_finite_touchedSuperT uniform_def by blast

(* Symmetry follows by normal induction: *)
lemma reneqv_sym:
"reneqv e e' ⟹ reneqv e' e"
apply(induct rule: reneqv.induct) 
by (auto intro!: reneqv.intros)  

lemma uniform_def2: "uniform e ⟷ (∃e'. reneqv e' e)"
unfolding uniform_def using reneqv_sym by auto

(* But to prove transitivity we will need inversion rules, which for the lambda case
will require (1) the custom presSuper equivariance and (2) a custom supervariable-based injectivity for 
lambda.  *)


lemma reneqv_iVr_casesL:
"reneqv (iVr x) e' ⟹ 
 (∃ xs x'. e' = iVr x' ∧ super xs ∧ {x,x'} ⊆ dsset xs)"
apply(erule reneqv.cases) by auto

lemma reneqv_iVr_casesR:
"reneqv e (iVr x') ⟹ 
 (∃ xs x. e = iVr x ∧ super xs ∧ {x,x'} ⊆ dsset xs)"
apply(erule reneqv.cases) by auto



lemma uniform_iLm_inject_super: 
assumes u: "uniform (iLm xs e)" and eq: "iLm xs e = iLm xs' e'" and super: "super xs" "super xs'"
shows "∃f. bij f ∧ |supp f| <o |UNIV::ivar set| ∧ presSuper f ∧ finite (touchedSuper (supp f)) ∧
       id_on (ILC.iFV (iLm xs e)) f ∧ id_on (- (dsset xs ∪ dsset xs')) f ∧ 
       id_on (dsset xs) (f o f) ∧ dsmap f xs = xs' ∧ irrename f e = e'"
apply(rule iLm_inject_super) using assms  
by (metis finite_Diff2 finite_singleton touchedSuper_iLm uniform_finite_touchedUponT, auto)


lemma reneqv_iLm_casesL:
assumes xs: "super xs" and rr: "reneqv (iLm xs e) ee'"
shows "∃ e'. ee' = iLm xs e' ∧ reneqv e e'"
proof-
  obtain xsa ea ea' where il: "iLm xs e = iLm xsa ea" and ee': "ee' = iLm xsa ea'" 
  and xsa: "super xsa" and r: "reneqv ea ea'" 
  using rr by cases auto

  have u: "uniform (iLm xsa ea)" using rr unfolding uniform_def ee' il by auto
  
  obtain f where f: "bij f" "|supp f| <o |UNIV::ivar set|" "presSuper f" 
  "id_on (iFV (iLm xsa ea)) f" "id_on (- (dsset xsa ∪ dsset xs)) f" 
  and xsaa: "dsmap f xsa = xs" and e: "e = irrename f ea"
  using uniform_iLm_inject_super[OF u il[symmetric] xsa xs] by auto

  have ff: "f ` (dsset xsa) = dsset xs" by (simp add: f(1) f(2) xsaa[symmetric])

  have "iFV (iLm xs e) ∩ dsset xs = {}" by auto
  hence "iFV ee' ∩ dsset xs = {}"  
  using reneqv_touchedSuperT_eq[OF rr] xs unfolding touchedSuperT_def touchedSuper_def by auto
  hence "iFV ea' ∩ dsset xs ⊆ dsset xsa" unfolding ee' by auto
  hence fv: "iFV ea' ∩ dsset xs = {} ∨ xs = xsa"  
  using super_disj[OF xs xsa] by auto

  show ?thesis apply(rule exI[of _ "irrename f ea'"]) apply(intro conjI)
    subgoal unfolding ee' unfolding iterm.inject apply(rule exI[of _ f]) apply(intro conjI)
      subgoal by fact
      subgoal by fact
      subgoal using f(5) fv unfolding id_on_def by auto
      subgoal by fact
      subgoal .. . 
    subgoal unfolding e using f(1) f(2) f(3) r irrename_reneqv by blast . 
qed

lemma reneqv_iLm_casesR:
assumes xs: "super xs'" and rr: "reneqv ee (iLm xs' e')"
shows "∃ e. ee = iLm xs' e ∧ reneqv e e'"
using reneqv_iLm_casesL reneqv_sym rr xs by blast

lemma reneqv_iLm_iff:
assumes "super xs"
shows "reneqv (iLm xs e) (iLm xs e') ⟷ reneqv e e'"
by (metis assms iLm_same_inject reneqv.iLm reneqv_iLm_casesR)

lemma reneqv_iApp_casesL:
assumes rr: "reneqv (iAp e1 es2) ee'"
shows "∃ e1' es2'. ee' = iAp e1' es2' ∧ reneqv e1 e1' ∧ 
          (∀e e'. {e,e'} ⊆ sset es2 ∪ sset es2' ⟶ reneqv e e')"
using assms by cases auto

lemma reneqv_iApp_casesR:
assumes rr: "reneqv ee (iAp e1' es2')"
shows "∃ e1 es2. ee = iAp e1 es2 ∧ reneqv e1 e1' ∧ 
          (∀e e'. {e,e'} ⊆ sset es2 ∪ sset es2' ⟶ reneqv e e')"
using assms by cases auto

lemma reneqv_iApp_iff:
"reneqv (iAp e1 es2) (iAp e1' es2') ⟷ 
 reneqv e1 e1' ∧ (∀e e'. {e,e'} ⊆ sset es2 ∪ sset es2' ⟶ reneqv e e')"
apply(subst reneqv.simps) by auto

(* goes by normal induction, once we have the inversion rules: *)
lemma reneqv_trans:
"reneqv e e' ⟹ reneqv e' e'' ⟹ reneqv e e''"
proof(induct arbitrary: e'' rule: reneqv.induct)
  case (iVr xs x x')
  then show ?case using reneqv_iVr_casesL[OF iVr(3)]
  using super_disj by (auto intro!: reneqv.iVr) 
next
  case (iLm xs e e')
  then show ?case using reneqv.iLm reneqv_iLm_casesL by blast
next
  case (iAp e1 e1' es2 es2')
  obtain e1'' es2'' where e'': "e'' = iAp e1'' es2''" 
  and 1: "reneqv e1' e1''" 
  and 2: "(∀e e'. {e, e'} ⊆ sset es2' ∪ sset es2'' ⟶ reneqv e e')"
  using reneqv_iApp_casesL[OF iAp(4)] by auto
  show ?case unfolding e'' apply(rule reneqv.iAp)
    subgoal using iAp.hyps(2) 1 by blast
    subgoal using iAp.hyps(3) 2  
    by auto (meson reneqv_sym snth_sset)+ .
qed 

lemma uniform_def3: "uniform e ⟷ reneqv e e"
using reneqv_trans reneqv_sym uniform_def by blast

corollary irrename_uniform:
assumes f: "bij f" "|supp f| <o |UNIV::ivar set|" "presSuper f"
and r: "uniform (e::itrm)" 
shows "uniform (irrename f e)"
using assms unfolding uniform_def3 
by (intro irrename_reneqv) auto

(* *)  (* requires strong induction: *)
lemma reneqv_itvsubst:
  assumes r: "reneqv e e'" and rr: "⋀xs x x'. super xs ⟹ {x, x'} ⊆ dsset xs ⟶ reneqv (f x) (f' x')"
    and s: "|SSupp f| <o |UNIV::ivar set|" "|SSupp f'| <o |UNIV::ivar set|"
    and f: "finite (touchedSuper (IImsupp f))" "finite (touchedSuper (IImsupp f'))"
  shows "reneqv (itvsubst f e) (itvsubst f' e')"
using r proof (binder_induction e e' avoiding: "IImsupp f" "IImsupp f'" rule: strong_induct_reneqv'')
  case bsmall
  then show ?case using f unfolding bsmall_def by (simp add: touchedSuper_Un)
next
  case Bound1
  then show ?case using s ILC.SSupp_IImsupp_bound by auto
next
  case Bound2
  then show ?case using s ILC.SSupp_IImsupp_bound by auto
next
  case (iVr xs x x')
  then show ?case using s rr by auto
next
  case (iLm ea e'a xs)
  then show ?case using iLm apply(subst iterm.subst)
    subgoal using s by auto
    subgoal using s by auto
    apply(subst iterm.subst)
    subgoal using s by auto
    subgoal using s by auto
    subgoal apply(rule reneqv.iLm) by auto .
next
  case (iAp e1 e1' es2 es2')
  then show ?case using rr
    apply(subst iterm.subst)
    subgoal using s by auto
    apply(subst iterm.subst)
    subgoal using s by auto
    subgoal apply(rule reneqv.iAp) apply auto
      by (meson reneqv_trans reneqv_sym)+ .
qed

definition "reneqvS es es' ≡ ∀e e'. {e,e'} ⊆ sset es ∪ sset es' ⟶ reneqv e e'"

lemma reneweqvS_sym:
"reneqvS es es' ⟹ reneqvS es' es"
unfolding reneqvS_def by auto

lemma reneweqvS_trans:
"reneqvS es es' ⟹ reneqvS es' es'' ⟹ reneqvS es es''"
unfolding reneqvS_def  using reneqv_trans shd_sset by auto blast+


definition "uniformS es ≡ ∃es'. reneqvS es es'"

lemma uniformS_def2: "uniformS es ⟷ (∃es'. reneqvS es' es)"
unfolding uniformS_def using reneweqvS_sym by blast

lemma uniformS_def3: "uniformS es ⟷ reneqvS es es"
unfolding uniformS_def using reneweqvS_sym reneweqvS_trans by blast

lemma uniformS_def4: 
"uniformS es = (∀e e'. {e, e'} ⊆ sset es ∪ sset es ⟶ reneqv e e')"
using uniformS_def3[unfolded reneqvS_def] .

lemma uniformS_smap2_iApp_iff: 
"uniformS (smap2 iAp es ess) ⟷ uniformS es ∧ uniformS (sflat ess)"
unfolding uniformS_def4 sset_sflat sset_smap2 unfolding sset_range apply auto
  apply (metis iterm.inject(2) reneqv_iApp_casesR)
  subgoal for i j i' j' apply(erule allE[of _ "iAp (es !! i) (ess !! i)"])
  apply(erule allE[of _ "iAp (es !! i') (ess !! i')"]) apply auto
  using reneqv_iApp_casesR by fastforce
  subgoal by (metis Un_iff insert_subset rangeI reneqv.iAp sset_range) .

lemma uniformS_smap2_iLm_iff:
assumes "super xs"
shows "uniformS (smap (iLm xs) es) ⟷ uniformS es"
using assms unfolding uniformS_def4 by (force simp: image_def reneqv_iLm_iff) 

lemma uniformS_sset_uniform: "uniformS es ⟹ e ∈ sset es ⟹ uniform e"
using reneqvS_def uniformS_def2 uniform_def3 by auto

lemma uniformS_touchedSuperT_eq: 
"uniformS es ⟹ {e,e'} ⊆ sset es ⟹ touchedSuperT e = touchedSuperT e'"
using reneqvS_def reneqv_touchedSuperT_eq uniformS_def3 by auto

lemma uniformS_touchedSuper_eq: 
"uniformS es ⟹ {e,e'} ⊆ sset es ⟹ touchedSuper (iFV e) = touchedSuper (iFV e')"
using uniformS_touchedSuperT_eq touchedSuperT_def by auto

lemma uniformS_touchedSuper: 
"uniformS es ⟹ e ∈ sset es ⟹ touchedSuper (⋃ (iFV ` (sset es))) = touchedSuper (iFV e)"
unfolding touchedSuper_UN  
by auto (metis empty_subsetI insert_subset touchedSuperT_def uniformS_touchedSuperT_eq)

lemma uniformS_touchedSuper_IImsupp_imkSubst: 
"super xs ⟹ uniformS es ⟹ e ∈ sset es ⟹ touchedSuper (IImsupp (imkSubst xs es)) ⊆ 
 {xs} ∪ touchedSuper (iFV e)"
using touchedSuper_IImsupp_imkSubst uniformS_touchedSuper by blast

lemma uniformS_touchedSuper_IImsupp_imkSubst':
"super xs ⟹ uniformS es ⟹ e ∈ sset es ⟹ 
  ys ≠ xs ⟹ ys ∉ touchedSuper (ILC.iFV e) ⟹ 
  ys ∉ touchedSuper (ILC.IImsupp (imkSubst xs es))"
using uniformS_touchedSuper_IImsupp_imkSubst by auto

lemma uniformS_touchedSuper_IImsupp_imkSubst'':
"super xs ⟹ super ys ⟹ uniformS es ⟹ e ∈ sset es ⟹ 
  ys ≠ xs ⟹ dsset ys ∩ ILC.iFV e = {} ⟹ 
  dsset ys ∩ ILC.IImsupp (imkSubst xs es) = {}"
using uniformS_touchedSuper_IImsupp_imkSubst' unfolding touchedSuper_def by blast

lemma super_uniformS_finite_touchedSuper_imkSubst: 
"super xs ⟹ uniformS es ⟹ finite (touchedSuper (IImsupp (imkSubst xs es)))"
by (metis finite_insert insert_is_Un rev_finite_subset snth_sset touchedSuperT_def 
touchedSuper_IImsupp_imkSubst uniformS_sset_uniform uniformS_touchedSuper uniform_finite_touchedUponT)

(* Mazza Lemma 11: *)
lemma reneqv_imkSubst:
assumes r: "reneqv e e'" and xs: "super xs" and rr: "reneqvS es es'" 
shows "reneqv (itvsubst (imkSubst xs es) e) (itvsubst (imkSubst xs es') e')"
apply(rule reneqv_itvsubst[OF r])
  subgoal by (smt (verit, del_insts) Un_iff imkSubst_def inf.absorb_iff2 inf_bot_right 
     insert_subset reneqv.simps reneqvS_def rr singleton_insert_inj_eq snth_sset touchedSuper_iVr xs)
  subgoal by simp
  subgoal by simp
  subgoal using rr super_uniformS_finite_touchedSuper_imkSubst uniformS_def xs by blast
  subgoal using rr super_uniformS_finite_touchedSuper_imkSubst uniformS_def2 xs by blast .

(* Mazza Lemma 12: *)
lemma uniform_imkSubst:
assumes u: "uniform e" and xs: "super xs" and rr: "uniformS es" 
shows "uniform (itvsubst (imkSubst xs es) e)"
using reneqv_imkSubst rr u uniformS_def3 uniform_def3 xs by blast


(* Inversion/simplification rules for "uniform": *)

lemma uniform_iVr[simp]:
"uniform (iVr x) ⟷ (∃xs. super xs ∧ x ∈ dsset xs)"
unfolding uniform_def3 by (meson bot.extremum iVr insert_subset reneqv_iVr_casesR)

lemma uniform_iLm_iff[simp]:
assumes xs: "super xs" 
shows "uniform (iLm xs e) ⟷ uniform e"
unfolding uniform_def3  
by (meson iLm reneqv_iLm_casesL reneqv_trans reneqv_sym xs)

(* It is impossible to express uniformity in terms of itself alone, 
this is why renaming equivalence was necessary... same as with parametricity etc.*)
lemma uniform_iApp_iff:
"uniform (iAp e es) ⟷ 
 uniform e ∧ (∀e e'. {e,e'} ⊆ sset es ⟶ reneqv e e')"
unfolding uniform_def3 using iAp reneqv_iApp_casesL by fastforce

lemma uniform_iApp_case:
"uniform (iAp e es) ⟹
 uniform e ∧ (∀e' ∈ sset es. uniform e')"
unfolding uniform_iApp_iff unfolding uniform_def3 by auto

(* Other properties: *)

lemma uniform_iLm_imp: "uniform (iLm xs e) ⟹ ∃ xs' e'. super xs' ∧ uniform e' ∧ iLm xs e = iLm xs' e'"
by (smt (verit, del_insts) iterm.distinct(5) iterm.distinct(6) reneqv.simps uniform_def2)

lemma super_bsmall: "super xs ⟹ bsmall (dsset xs)"
by (metis bsmall_def finite.emptyI finite_insert super_dsset_singl touchedSuper_def)

lemma uniform_iLm_imp_avoid: 
assumes u: "uniform (iLm xs e)" and A: "small A" "bsmall A"
shows "∃ xs' e'. super xs' ∧ uniform e' ∧ iLm xs e = iLm xs' e' ∧ dsset xs' ∩ A = {}"
proof-
  obtain xs' e' where xs': "super xs'" and e': "uniform e'" and il: "iLm xs e = iLm xs' e'"
  using uniform_iLm_imp[OF u] by auto
  define B where B: "B = A ∪ dsset xs' ∪ iFV (iLm xs e)"
  have bsB: "bsmall B" unfolding B apply(intro ILC_Renaming_Equivalence.bsmall_Un)
    subgoal by fact
    subgoal using super_bsmall[OF xs'] .
    subgoal using u bsmall_def touchedSuperT_def uniform_finite_touchedUponT by fastforce .
  hence BB: "finite (touchedSuper B)" unfolding bsmall_def by auto
  have BBB: "|B| <o |UNIV::ivar set|"  
    by (metis B small_def assms(2) card_dsset_ivar iterm.set_bd_UNIV var_stream_class.Un_bound)
  obtain xs'' where xxs'': "super xs''" "B ∩ dsset xs'' = {}" 
    by (smt (verit) Collect_cong Int_commute bsB bsmall_def super_infinite touchedSuper_def)
  obtain f where xs'': "xs'' = dsmap f xs'" and f: "bij_betw f (dsset xs') (dsset xs'')" "id_on (- dsset xs') f" 
  by (metis ex_dsmap)
  obtain g where g: "bij g" "|supp g| <o |UNIV::ivar set|"
      "presSuper g" "g ` dsset xs' ∩ B = {}" "id_on (- dsset xs' ∩ - dsset xs'') g" 
      "eq_on (dsset xs') g f"
  using extend_super1[OF xs' xxs''(1) BBB BB xxs''(2), of "{}", simplified, OF f(1) xs''[symmetric]]
  by auto

  
  show ?thesis apply(intro exI[of _ xs''] exI[of _ "irrename g e'"], safe)
    subgoal by fact
    subgoal using e' g(1) g(2) g(3) irrename_uniform by presburger
    subgoal unfolding il xs'' iterm.inject apply(rule exI[of _ g], safe)
      subgoal by fact subgoal by fact
      subgoal using g(5) unfolding id_on_def apply simp  
        by (metis B DiffI UnCI disjoint_iff il iterm.set(3) xxs''(2))
      subgoal apply(rule dsmap_cong)
        subgoal by (simp add: g(1) inj_on_def bij_implies_inject)
        subgoal using bij_betw_def f(1) by blast
        subgoal by (meson eq_on_def g(6)) . .
    subgoal using B xxs''(2) by blast . 
qed
     
lemma iLm_switch_to_super: 
assumes u: "uniform (iAp (iLm xs e1) es2)"
shows 
"∃xs' e1'. super xs' ∧ iLm xs e1 = iLm xs' e1' ∧ 
           itvsubst (imkSubst xs es2) e1 = itvsubst (imkSubst xs' es2) e1' ∧ 
           (∀e∈sset es2. dsset xs' ∩ iFV e = {})"
proof- 
  obtain e2 where e2: "e2 ∈ sset es2" 
    using snth_sset by blast
  
  have uu: "⋀e. e ∈ sset es2 ⟹ uniform e"
  using u uniform_iApp_case by blast
  have u': "uniform (iLm xs e1)" using u uniform_iApp_iff by blast

  have fves2: "touchedSuper (⋃ (iFV ` (sset es2))) = touchedSuper (iFV e2) "  
    by (metis e2 iterm.inject(2) reneqvS_def reneqv_iApp_casesL u uniformS_def3 uniformS_touchedSuper uniform_def3)

  have ss: "small (⋃ (iFV ` (sset es2)))" 
  unfolding small_def apply(rule iterm.UN_bound)
    subgoal by (simp add: countable_card_ivar countable_sset)
    subgoal using iterm.set_bd_UNIV by blast .

  have bs: "bsmall (⋃ (iFV ` (sset es2)))"
  unfolding bsmall_def fves2 unfolding bsmall_def[symmetric]  
  using bsmall_def e2 touchedSuperT_def uniform_finite_touchedUponT uu by presburger

  obtain xs' e1' where xs': "super xs'" and e1': "uniform e1'" and il: "iLm xs e1 = iLm xs' e1'" 
   "dsset xs' ∩ (⋃ (iFV ` (sset es2))) = {}"
  using bs ss u' uniform_iLm_imp_avoid by blast

  obtain f where f: "bij f" "|supp f| <o |UNIV::ivar set|" "id_on (ILC.iFV (iLm xs e1)) f" 
  "dsmap f xs = xs'" "irrename f e1 = e1'" using il[unfolded iterm.inject] by auto
  
  show ?thesis apply(intro exI[of _ xs'] exI[of _ "irrename f e1"], safe)
    subgoal by fact
    subgoal unfolding il f ..
    subgoal apply(subst irrename_eq_itvsubst_iVr')
      subgoal by fact
      subgoal by fact
      subgoal apply(subst itvsubst_comp)
        subgoal by simp
        subgoal using f(2) by auto
        subgoal apply(rule itvsubst_cong)
          subgoal by simp
          subgoal by (simp add: SSupp_itvsubst_bound f(2))
          subgoal apply simp unfolding f(4)[symmetric] 
            by (metis (full_types) Diff_iff f(1) f(3) f(4) id_on_def il(1) imkSubst_def 
            imkSubst_smap iterm.set(3)) . . .
     subgoal using il(2) by auto .
qed
          
lemma reneqv_head_reduction: 
assumes r: "reneqv (iAp (iLm xs e1) es2) (iAp (iLm ys d1) ds2)"
shows "reneqv (itvsubst (imkSubst xs es2) e1) (itvsubst (imkSubst ys ds2) d1)"
proof-  (* we needed some delicate renaming lemmas, to turn 
   the bound sequences into supervariables, to allow for the nice inversion rules 
  for renaming equivalence to pop up... *)
  have uu: "uniform (iAp (iLm xs e1) es2)" "uniform (iAp (iLm ys d1) ds2)"
  using r using uniform_def uniform_def2 by auto
  obtain xs' e1' where x0: "super xs'" and x1: "iLm xs e1 = iLm xs' e1'"
         and x2: "itvsubst (imkSubst xs es2) e1 = itvsubst (imkSubst xs' es2) e1'"
         and x3: "∀e∈sset es2. dsset xs' ∩ iFV e = {}"
  using uu iLm_switch_to_super by fastforce
  hence ux': "uniform (iAp (iLm xs' e1') es2)"  
  using uu by auto
  obtain ys' d1' where y0: "super ys'" and y1: "iLm ys d1 = iLm ys' d1'"
         and y2: "itvsubst (imkSubst ys ds2) d1 = itvsubst (imkSubst ys' ds2) d1'"
         and y3: "∀e∈sset ds2. dsset ys' ∩ iFV e = {}"
  using uu iLm_switch_to_super by fastforce
  hence uy': "uniform (iAp (iLm ys' d1') ds2)"  
  using uu by auto
  have r: "reneqv (iAp (iLm xs' e1') es2) (iAp (iLm ys' d1') ds2)"
  using r x1 y1 by auto 
  hence "reneqv (iLm xs' e1') (iLm ys' d1')" and rr: "∀e e'. {e,e'} ⊆ sset es2 ∪ sset ds2 ⟶ reneqv e e'"
  using reneqv_iApp_casesR by fastforce+
  then obtain ee1' where il: "iLm ys' d1' = iLm xs' ee1'" and r: "reneqv e1' ee1'"
  using reneqv_iLm_casesL x0 by blast
  hence yy1: "iLm ys d1 = iLm xs' ee1'" using y1 by simp

  have itv: "itvsubst (imkSubst xs es2) e1 = itvsubst (imkSubst xs' es2) e1'"
  using iLm_eq_imkSubst[OF x1] .
  have iitv: "itvsubst (imkSubst ys ds2) d1 = itvsubst (imkSubst xs' ds2) ee1'"
  using iLm_eq_imkSubst[OF yy1] . 
  
  show ?thesis unfolding itv iitv apply(rule reneqv_imkSubst)
  using r x0 rr unfolding reneqvS_def by auto
qed

lemma uniform_head_reduction: 
assumes u: "uniform (iAp (iLm xs e1) es2)"
shows "uniform (itvsubst (imkSubst xs es2) e1)"
using assms unfolding uniform_def3 by (elim reneqv_head_reduction)
