Theory Translation_ILC_to_LC

(* The translations back and forth between the infinitary and finitary lambda-calculi *)
theory Translation_ILC_to_LC
imports ILC_relations_more Super_Recursor
begin

(* Translating (affine, uniform) finitary lambda to finitary-lambda *)


(* *)
definition B :: "lterm set" where "B ≡ UNIV"


(* Restricting a (supervariable-preserving) ivar-permutation to a ver-permutation *)
definition restr :: "(ivar ⇒ ivar) ⇒ var ⇒ var" where 
"restr f x ≡ subOf (dsmap f (superOf x))"

lemma restr_id[simp]: "restr id = id"
unfolding restr_def by auto

lemma restr_comp: "bij f ⟹ |supp f| <o |UNIV::ivar set| ⟹ presSuper f ⟹ 
bij g ⟹ |supp g| <o |UNIV::ivar set| ⟹ presSuper g ⟹ 
restr (g o f) = restr g o restr f"
unfolding restr_def fun_eq_iff 
apply(subst dstream_map_comp[symmetric]) 
by (auto simp add: presSuper_def)

lemma bij_restr: 
assumes f: "bij f" "|supp f| <o |UNIV::ivar set|" "presSuper f"
shows "bij (restr f)"
proof-
  have 0: "⋀d. dsmap f (dsmap (inv f) d) = d" 
  using f(1,2) by (simp add: dstream.map_comp)
  show ?thesis unfolding bij_def[of "restr f"] inj_def surj_def restr_def apply safe
    subgoal using f by (metis bij_imp_bij_inv dstream.map_comp dstream.map_id inv_o_simp1 
       presSuper_def subOf_superOf superOf_subOf super_superOf supp_inv_bound)
    subgoal for y by (metis 0 f(3) presSuper_def subOf_inj subOf_superOf super_superOf) .
qed

term "subOf ` (touchedSuper (supp f))"

lemma supp_restr: 
assumes f: "bij f" "|supp f| <o |UNIV::ivar set|" 
shows "supp (restr f) ⊆ subOf ` (touchedSuper (supp f))"
unfolding restr_def supp_def image_def touchedSuper_def 
by auto (smt (verit, ccfv_threshold) disjoint_iff dstream_map_ident_strong mem_Collect_eq subOf_superOf super_superOf)

lemma card_supp_restr: 
assumes f: "bij f" "|supp f| <o |UNIV::ivar set|" "bsmall (supp f)" "presSuper f"
shows "|supp (restr f)| <o |UNIV::var set|" 
proof-
  have "|supp (restr f)| ≤o |subOf ` (touchedSuper (supp f))|"
  using supp_restr[OF f(1,2)] by auto
  also have "|subOf ` (touchedSuper (supp f))| <o |UNIV::var set|"
  using f(3) unfolding bsmall_def by (simp add: finite_card_var)
  finally show ?thesis .
qed

lemma finite_supp_restr: 
assumes "bij f" "|supp f| <o |UNIV::ivar set|" "bsmall (supp f)" "presSuper f"
shows "finite (supp (restr f))" 
using card_supp_restr[OF assms] unfolding finite_iff_le_card_var .

lemma restr_cong_id: 
assumes "bij f" "|supp f| <o |UNIV::ivar set|" "bsmall (supp f)" "presSuper f"
and "⋀y x. y ∈ FFVars e ⟹ x ∈ dsset (superOf y) ⟹ f x = x" "z ∈ LC.FFVars e"
shows "restr f z = z"
using assms unfolding restr_def by (simp add: dstream_map_ident_strong)


(* *)

definition iVrB where "iVrB x ≡ Vr (subOf (fst (theSN x)))"
definition iLmB where "iLmB (xs::ivar dstream) b ≡ Lm (subOf xs) b"
definition iApB where "iApB b1 bs2 ≡ Ap b1 (snth bs2 0)"
definition renB where "renB f b ≡ rrename (restr f) b"
definition FVarsB where "FVarsB b ≡ ⋃ ((dsset o superOf) ` (FFVars b))"


lemma iVrB_B: "super xs ⟹ x ∈ dsset xs ⟹ iVrB x ∈ B"
unfolding B_def by auto

lemma iApB_B: "b1 ∈ B ⟹ sset bs2 ⊆ B ⟹ iApB b1 bs2 ∈ B"
unfolding B_def by auto

lemma iLmB_B: "b ∈ B ⟹ super xs ⟹ iLmB xs b ∈ B"
unfolding B_def by auto

lemma renB_B: "bij σ ⟹ |supp σ| <o |UNIV::ivar set| ⟹ bsmall (supp σ) ⟹ presSuper σ ⟹ 
  b ∈ B ⟹ renB σ b ∈ B"
unfolding B_def by auto

lemma renB_id: "b ∈ B ⟹ renB id b = b"
unfolding renB_def by auto

lemma renB_comp: "bij σ ⟹ |supp σ| <o |UNIV::ivar set| ⟹ bsmall (supp σ) ⟹ presSuper σ ⟹ 
    bij τ ⟹ |supp τ| <o |UNIV::ivar set| ⟹ bsmall (supp τ) ⟹ presSuper τ ⟹ 
    b ∈ B ⟹ renB (τ o σ) b = renB τ (renB σ b)"
unfolding renB_def apply(subst restr_comp) 
  by (auto simp add: bij_restr card_supp_restr ltermP.rrename_comps)

lemma renB_cong: "bij σ ⟹ |supp σ| <o |UNIV::ivar set| ⟹ bsmall (supp σ) ⟹ presSuper σ ⟹ 
   (∀xs ∈ touchedSuper (FVarsB b). dsmap σ xs = xs) ⟹ 
   renB σ b = b"
unfolding renB_def FVarsB_def apply(rule ltermP.rrename_cong_ids)
by (auto simp: bij_restr card_supp_restr restr_def touchedSuper_UN intro: restr_cong_id)

lemma renB_FVarsB: "bij σ ⟹ |supp σ| <o |UNIV::ivar set| ⟹ bsmall (supp σ) ⟹ presSuper σ ⟹ 
   x ∈ FVarsB (renB σ b) ⟷ inv σ x ∈ FVarsB b"
unfolding renB_def FVarsB_def apply safe
  subgoal by simp (metis (no_types, lifting) bij_restr card_supp_restr dstream.set_map image_in_bij_eq inv_simp2 
    presSuper_def restr_def superOf_subOf super_superOf ltermP.FFVars_rrenames)
  subgoal by simp (metis bij_restr card_supp_restr dstream.set_map image_in_bij_eq inv_simp1 presSuper_def 
    restr_def superOf_subOf super_superOf ltermP.FFVars_rrenames) .

lemma renB_iVrB[simp]: "bij σ ⟹ |supp σ| <o |UNIV::ivar set| ⟹ bsmall (supp σ) ⟹ presSuper σ ⟹ 
  super xs ⟹ x ∈ dsset xs ⟹ 
  renB σ (iVrB x) = iVrB (σ x)"
unfolding renB_def iVrB_def apply(subst ltermP.permute)
  subgoal by (auto simp add: bij_restr)
  subgoal by (auto simp add: card_supp_restr)
  subgoal unfolding restr_def apply(cases "theSN x", cases "theSN (σ x)") 
  by simp (smt (verit, ccfv_SIG) RSuper_def UN_iff bij_betw_def bij_betw_inv_into bij_superOf 
     dstream.set_map dtheN image_in_bij_eq inv_simp2 mem_Collect_eq presSuper_def superOf_subOf 
     theSN_unique) .

lemma renB_iApB[simp]: "bij σ ⟹ |supp σ| <o |UNIV::ivar set| ⟹ bsmall (supp σ) ⟹ presSuper σ ⟹ 
   b1 ∈ B ⟹ sset bs2 ⊆ B ⟹
   renB σ (iApB b1 bs2) = iApB (renB σ b1) (smap (renB σ) bs2)"
unfolding renB_def iApB_def apply(subst ltermP.permute)
  subgoal by (auto simp add: bij_restr)
  subgoal by (auto simp add: card_supp_restr)
  subgoal by auto .

lemma renB_iLmB[simp]: "bij σ ⟹ |supp σ| <o |UNIV::ivar set| ⟹ bsmall (supp σ) ⟹ presSuper σ ⟹ 
   b ∈ B ⟹ super xs ⟹ 
   renB σ (iLmB xs b) = iLmB (dsmap σ xs) (renB σ b)"
unfolding renB_def iLmB_def apply(subst ltermP.permute)
  subgoal by (auto simp add: bij_restr)
  subgoal by (auto simp add: card_supp_restr)
  subgoal using restr_def superOf_subOf by auto .

lemma FVarsB_iVrB: "super xs ⟹ x ∈ dsset xs ⟹ touchedSuper (FVarsB (iVrB x)) ⊆ touchedSuper {x}"
unfolding FVarsB_def iVrB_def apply(cases "theSN x") 
  by auto (metis (mono_tags, lifting) Int_emptyD dtheN insert_subset mem_Collect_eq mk_disjoint_insert 
   singletonI superOf_subOf super_dsset_RSuper theSN_unique touchedSuper_def)

(* Unlike FVarsB_iVrB, we have that 
FVarsB_iApB and FVarsB_iLmB actually even hold in a "raw", stronger version, with touchedSuper removed and with 
FVarsB_iLmB formulated as follows: "b ∈ B ⟹ super xs ⟹ FVarsB (iLmB xs b) ⊆ FVarsB b - dsset xs" *)
lemma FVarsB_iApB: "b1 ∈ B ⟹ sset bs2 ⊆ B ⟹ 
 touchedSuper (FVarsB (iApB b1 bs2)) ⊆ 
 touchedSuper (FVarsB b1) ∪ ⋃ ((touchedSuper o FVarsB) ` (sset bs2))"
unfolding FVarsB_def iApB_def touchedSuper_def by (fastforce simp: shd_sset)

lemma FVarsB_iLmB: "b ∈ B ⟹ super xs ⟹
  touchedSuper (FVarsB (iLmB xs b)) ⊆ touchedSuper (FVarsB b) - {xs}"
unfolding FVarsB_def iLmB_def touchedSuper_def
by auto (metis Int_emptyD subOf_superOf super_disj super_superOf)

interpretation T' : ILC_SuperRec where 
B = B and iVrB = iVrB and iApB = iApB and iLmB = iLmB and renB = renB and FVarsB = FVarsB
apply standard
using iVrB_B iApB_B iLmB_B renB_B renB_id renB_comp 
renB_iVrB renB_iApB renB_iLmB
FVarsB_iVrB FVarsB_iApB FVarsB_iLmB apply auto  
by (auto simp add: renB_cong renB_FVarsB)  


(* END PRODUCT: *)

definition tr' :: "ilterm ⇒ lterm" where "tr' = T'.rec"

lemma tr'_iVr[simp]: "super xs ⟹ x ∈ dsset xs ⟹ tr' (iVr x) = Vr (subOf (fst (theSN x)))"
using T'.rec_iVr unfolding tr'_def iVrB_def by auto

lemma tr'_iLm[simp]: "super xs ⟹ good e ⟹ tr' (iLm xs e) = Lm (subOf xs) (tr' e)"
using T'.rec_iLm unfolding tr'_def iLmB_def by auto

lemma tr'_iAp[simp]: "good e1 ⟹ (∀e2∈sset es2. good e2) ⟹ 
  (∀e2 e2'. {e2,e2'} ⊆ sset es2 ⟶ touchedSuperT e2 = touchedSuperT e2') ⟹ 
  tr' (iAp e1 es2) = Ap (tr' e1) (tr' (snth es2 0))"
using T'.rec_iAp unfolding tr'_def iApB_def by auto

lemma irrename_tr':
"good e ⟹ bij f ⟹ |supp f| <o |UNIV::ivar set| ⟹ bsmall (supp f) ⟹ presSuper f ⟹
 tr' (irrename f e) = rrename (restr f) (tr' e)"
using T'.rec_irrename unfolding tr'_def renB_def by auto

lemma FFVars_tr': 
assumes "good e" "x ∈ LC.FFVars (tr' e)"
shows "dsset (superOf x) ∩ ILC.FFVars e ≠ {}"
proof-
  have "superOf x ∈ touchedSuper (ILC.FFVars e)"
  using assms using T'.FVarsB_rec  FVarsB_def unfolding tr'_def  by (auto simp: touchedSuper_Union)
  thus ?thesis unfolding touchedSuper_def by auto
qed

(* *)

lemma reneqv_good: "reneqv e e' ⟹ good e ∧ good e'"
apply(induct rule: reneqv.induct) 
  subgoal by(auto intro: good.intros) 
  subgoal by(auto intro: good.intros) 
  subgoal apply(rule conjI)
    subgoal apply(rule good.iAp) using reneqv_touchedSuperT_eq by blast+
    subgoal apply(rule good.iAp) using reneqv_touchedSuperT_eq by blast+ . .

lemma uniform_good: "uniform e ⟹ good e"
using reneqv_good unfolding uniform_def3 by auto

lemma uniformS_good: 
"uniformS es2 ⟹ 
 (∀e2∈sset es2. good e2) ∧ 
 (∀e2 e2'. {e2,e2'} ⊆ sset es2 ⟶ touchedSuperT e2 = touchedSuperT e2')"
unfolding uniformS_def4 using reneqv_good reneqv_touchedSuperT_eq by auto 


(* We recover Mazza's desired definition: *)

thm tr'_iVr 

lemma tr'_iLm_uniform[simp]: "super xs ⟹ uniform e ⟹ tr' (iLm xs e) = Lm (subOf xs) (tr' e)"
using uniform_good by auto

lemma tr'_iAp_uniform[simp]: "uniform e1 ⟹ uniformS es2 ⟹ 
  tr' (iAp e1 es2) = Ap (tr' e1) (tr' (snth es2 0))"
by (simp add: uniformS_good uniform_good)

lemma irrename_tr'_uniform:
"uniform e ⟹ bij f ⟹ |supp f| <o |UNIV::ivar set| ⟹ bsmall (supp f) ⟹ presSuper f ⟹
 tr' (irrename f e) = rrename (restr f) (tr' e)"
using uniform_good irrename_tr' by auto

lemma FFVars_tr'_uniform: 
assumes "uniform e" "x ∈ LC.FFVars (tr' e)"
shows "dsset (superOf x) ∩ ILC.FFVars e ≠ {}"
using assms FFVars_tr'  uniform_good by auto
