inductive step :: "trm ⇒ trm ⇒ bool" where
  Beta: "step (App (Lam x e1) e2) (tvsubst (Var(x:=e2)) e1)"
| AppL: "step e1 e1' ⟹ step (App e1 e2) (App e1' e2)"
| AppR: "step e2 e2' ⟹ step (App e1 e2) (App e1 e2')"
| Xi: "step e e' ⟹ step (Lam x e) (Lam x e')"

(* user goals *)
lemma perm_id0s:
  "rrename id = id"
  "rrename id = id"
  by (rule term.permute_id0)+

lemma perm_comps:
  fixes σ :: "var ⇒ var"
  assumes "bij σ" "|supp σ| <o |UNIV::var set|" "bij σ'" "|supp σ'| <o |UNIV::var set|"
  shows
    "rrename σ (rrename σ' x1) = rrename (σ ∘ σ') x1"
    "rrename σ (rrename σ' x2) = rrename (σ ∘ σ') x2"
  by (rule term.permute_comp assms)+

lemma perm_supports:
  fixes σ :: "var ⇒ var"
  assumes "bij σ" "|supp σ| <o |UNIV::var set|"
  shows
    "(⋀a. a ∈ FFVars x1 ⟹ σ a = a) ⟹ rrename σ x1 = x1"
    "(⋀a. a ∈ FFVars x2 ⟹ σ a = a) ⟹ rrename σ x2 = x2"
  by (rule term.permute_cong_id[OF assms], assumption)+

lemma supp_smalls:
  fixes x1 x2::"trm"
  shows
    "|FFVars x1| <o |UNIV::var set|"
    "|FFVars x2| <o |UNIV::var set|"
  by (rule term.set_bd_UNIV)+

lemma supp_seminat:
  fixes σ :: "var ⇒ var"
  assumes "bij σ" "|supp σ| <o |UNIV::var set|"
  shows
    "FFVars (rrename σ x1) ⊆ σ ` FFVars x1"
    "FFVars (rrename σ x2) ⊆ σ ` FFVars x2"
  using term.FVars_permute assms by blast+

thm step_def
definition "G ≡ λp (B::var set) x1 x2.
  (∃x e1 e2. B = {x} ∧ x1 = App (Lam x e1) e2 ∧ x2 = tvsubst (Var(x := e2)) e1) ∨
  (∃e1 e1' e2. B = {} ∧ x1 = App e1 e2 ∧ x2 = App e1' e2 ∧ p e1 e1') ∨
  (∃e2 e2' e1. B = {} ∧ x1 = App e1 e2 ∧ x2 = App e1 e2' ∧ p e2 e2') ∨
  (∃e e' x. B = {x} ∧ x1 = Lam x e ∧ x2 = Lam x e' ∧ p e e')"

lemma G_equiv:
  assumes "bij σ" "|supp σ| <o |UNIV::var set|"
  shows "G R B x1 x2 ⟹ G (λx1 x2. R (rrename (inv σ) x1) (rrename (inv σ) x2)) (σ ` B) (rrename σ x1) (rrename σ x2)"
  using assms apply -
  unfolding G_def
  apply (elim disj_forward) apply (auto simp: term.permute_comp rrename_tvsubst_comp)
  sledgehammer
  by (metis inv_simp1 term.permute_bij term.permute_inv_simp)

abbreviation "supp_T x1 x2 ≡ FFVars x1 ∪ FFVars x2"

lemma fresh: "∃xx. xx ∉ supp_T (x1::trm) x2"
  using Lam_avoid supp_smalls(2) term.set(2) by blast

lemma G_refresh:
  assumes
    "⋀x1 x2. R x1 x2 ⟹ step x1 x2"
    "⋀σ x1 x2. bij σ ⟹ |supp σ| <o |UNIV::var set| ⟹ R x1 x2 ⟹ R (rrename σ x1) (rrename σ x2)"
  shows "G R B x1 x2 ⟹ ∃B'. B' ∩ supp_T x1 x2 = {} ∧ G R B' x1 x2"
    unfolding ex_push_inwards conj_disj_distribL ex_disj_distrib G_def
    apply (elim disj_forward exE)
       apply auto
    using fresh[of x1 x2] assms(2) Lam_refresh apply auto
    by (metis Lam_eq_tvsubst singletonD)

lemma BE_iinduct:
  fixes K::"'p ⇒ var set"
  assumes smalls: "⋀p. |K p| <o |UNIV::var set|"
  and II: "step x1 x2"
  and step: "⋀B x1 x2 p. B ∩ K p = {} ⟹ B ∩ supp_T x1 x2 = {} ⟹
    G (λx1' x2'. step x1' x2' ∧ All (R x1' x2')) B x1 x2 ⟹ R x1 x2 p"
shows "R x1 x2 p"
  apply (rule BE_iinduct_aux[of K, unfolded II_eq_II'[symmetric] II_eq, OF smalls II,
    THEN spec, THEN mp[OF _ bij_id, THEN mp[OF _ supp_id_bound]], THEN spec,
    unfolded perm_ids
  ])
  apply (rule step)
  apply assumption+
  done
