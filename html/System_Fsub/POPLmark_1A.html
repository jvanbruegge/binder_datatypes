(********************* Actual formalization ************************)

declare ty.intros[intro]

lemma well_scoped:
  assumes "Γ ⊢ S <: T"
  shows "FFVars_sftypeP S ⊆ dom Γ" "FFVars_sftypeP T ⊆ dom Γ"
using assms proof (induction Γ S T rule: ty.induct)
  case (SA_Trans_TVar x U Γ T) {
  case 1 then show ?case using SA_Trans_TVar
    by (metis fst_conv imageI singletonD subsetI sftypeP.set(1))
next
  case 2 then show ?case using SA_Trans_TVar by simp
} qed auto

declare SA_Refl_TVar[intro!]

lemma ty_refl: "⟦wf Γ ; FFVars_sftypeP T ⊆ dom Γ ⟧ ⟹ Γ ⊢ T <: T"
proof (binder_induction T arbitrary: Γ avoiding: "dom Γ" rule: sftypeP.strong_induct)
  case (TVr x Γ)
  then show ?case by (simp add: SA_Refl_TVar)  
qed (auto simp: Diff_single_insert SA_All wf_Cons)

lemma ty_permute: "⟦ Γ ⊢ S <: T ; wf Δ ; set Γ = set Δ ⟧ ⟹ Δ ⊢ S <: T"
proof (binder_induction Γ S T arbitrary: Δ avoiding: "dom Δ" rule: ty.strong_induct)
  case (SA_Refl_TVar Γ x Δ)
  then show ?case by blast
next
  case (SA_All Γ T⇩1 S⇩1 x S⇩2 T⇩2 Δ)
  then have "set (Γ ,, x <: T⇩1) = set (Δ ,, x <: T⇩1)" by auto
  then show ?case by (meson SA_All ty.SA_All well_scoped(1) wf_Cons)
qed auto

lemma wf_concat: "⟦ wf Δ ; wf Γ ; Γ ⊥ Δ ⟧ ⟹ wf (Γ,,Δ)"
proof (induction Δ rule: wf.induct)
  case (wf_Cons x Δ T)
  then have 1: "(Γ,, (Δ ,, x <: T)) = ((Γ,, Δ),, x <: T)" by simp
  show ?case unfolding 1
    apply (rule wf.wf_Cons)
    using wf_Cons by auto
qed auto

lemma weaken_closed: "⟦ FFVars_sftypeP S ⊆ dom Γ ; Γ ⊥ Δ ⟧ ⟹ FFVars_sftypeP S ⊆ dom (Γ,,Δ)"
  by auto

lemma wf_concat_disjoint: "wf (Γ,, Δ) ⟹ Γ ⊥ Δ"
proof (induction Δ)
  case (Cons a Δ)
  then show ?case
    by (smt (verit, del_insts) Un_iff append_Cons disjoint_iff fst_conv image_iff inf.idem insertE list.inject list.simps(15) set_append set_empty2 wf.cases)
qed simp

lemma wf_insert: "⟦ wf (Γ,,Δ); x ∉ dom Γ ; x ∉ dom Δ ; FFVars_sftypeP T ⊆ dom Γ ⟧ ⟹ wf (Γ,,x<:T,,Δ)"
  by (induction Δ) auto

lemma ty_weakening:
  assumes "Γ ⊢ S <: T" "wf (Γ,,Δ)"
shows "Γ,,Δ ⊢ S <: T"
using assms proof (binder_induction Γ S T avoiding: "dom Δ" Γ S T rule: ty.strong_induct)
  case (SA_Top Γ S)
  then show ?case using ty.SA_Top weaken_closed wf_concat_disjoint by presburger
next
  case (SA_Refl_TVar Γ x)
  then show ?case using ty.SA_Refl_TVar weaken_closed wf_concat_disjoint  
    by (meson ty_refl well_scoped(1))   
next
  case (SA_All Γ T⇩1 S⇩1 x S⇩2 T⇩2)
  have 1: "wf (Γ,, x <: T⇩1,, Δ)"
    by (meson wf_insert SA_All.hyps(1,4,5,9) UnCI well_scoped(1))
  have 2: "wf (Γ ,, Δ ,, x <: T⇩1)"
    by (metis SA_All.hyps(1,4,6,9) UnE UnI1 image_Un set_append well_scoped(1) wf_Cons)
  show ?case using ty_permute[OF _ 2] 1 SA_All by auto
qed auto

corollary ty_weakening_extend: "⟦ Γ ⊢ S <: T ; X ∉ dom Γ ; FFVars_sftypeP Q ⊆ dom Γ ⟧ ⟹ Γ,,X<:Q ⊢ S <: T"
  using ty_weakening[of _ _ _ "[(X, Q)]"] by (metis append_Cons append_Nil wf_Cons wf_context)

lemma wf_concatD: "wf (Γ,, Δ) ⟹ wf Γ"
  by (induction Δ) auto

lemma context_determ: "⟦ X <: U ∈ (Γ ,, X <: U',, Δ) ;  wf (Γ ,, X <: U',, Δ) ⟧ ⟹ U = U'"
proof (induction Δ)
  case Nil
  then show ?case
    by (metis Pair_inject append_Nil fst_conv image_eqI set_ConsD wf_ConsE)
qed auto

lemma narrow_wf: "⟦ wf ((Γ ,, X <: Q),, Δ) ; FFVars_sftypeP R ⊆ dom Γ ⟧ ⟹ wf ((Γ,, X <: R),, Δ)"
proof (induction Δ)
  case (Cons a Δ)
  then have "wf (Γ,, X <: R,, Δ)" by auto
  obtain b c where 2: "a = (b, c)" by fastforce
  then have 1: "⋀R. Γ ,, X <: R,, (a # Δ) = (b, c) # (Γ ,, X <: R,, Δ)" by simp
  have "b ∉ dom (Γ,, X <: R,, Δ)" using Cons(2)[unfolded 1] by auto
  then show ?case unfolding 1 using Cons 2 by auto
qed auto

lemma SA_AllE1[consumes 2, case_names SA_Trans_TVar SA_All]:
  assumes "Γ ⊢ ∀X<:S⇩1. S⇩2 <: T" "X ∉ dom Γ"
    and Top: "⋀Γ. ⟦wf Γ;   FFVars_sftypeP (∀X<:S⇩1. S⇩2) ⊆ dom Γ ⟧ ⟹ R Γ (∀X<:S⇩1. S⇩2) Top"
    and Forall: "⋀Γ T⇩1 T⇩2. ⟦ Γ ⊢ T⇩1 <: S⇩1 ; Γ,, X<:T⇩1 ⊢ S⇩2 <: T⇩2 ⟧ ⟹ R Γ (∀X<:S⇩1. S⇩2) (∀X<:T⇩1 . T⇩2)"
  shows "R Γ (∀X<:S⇩1. S⇩2) T"
using assms(1,2) proof (binder_induction Γ "∀X<:S⇩1. S⇩2" T avoiding: Γ "∀X<:S⇩1. S⇩2" T rule: ty.strong_induct)
  case (SA_All Γ T⇩1 R⇩1 Y R⇩2 T⇩2)
  have 1: "∀Y<:T⇩1 . T⇩2 = ∀X<:T⇩1. rrename_sftypeP (id(Y:=X,X:=Y)) T⇩2"
    apply (rule Forall_swap)
    using SA_All(6,9) well_scoped(2) by fastforce
  have fresh: "X ∉ FFVars_sftypeP T⇩1"
    by (meson SA_All(4,9) in_mono well_scoped(1))
  have same: "R⇩1 = S⇩1" using SA_All(8) sftypeP_inject(3) by blast
  have x: "∀Y<:S⇩1. R⇩2 = ∀X<:S⇩1. rrename_sftypeP (id(Y:=X,X:=Y)) R⇩2"
    apply (rule Forall_swap)
    by (metis (no_types, lifting) SA_All(8) assms(1,2) in_mono sup.bounded_iff sftypeP.set(4) well_scoped(1))
  show ?case unfolding 1
    apply (rule Forall)
    using same SA_All(4) apply simp
    apply (rule iffD2[OF arg_cong3[OF _ _ refl, of _ _ _ _ ty], rotated -1])
      apply (rule ty.equiv)
       apply (rule bij_swap supp_swap_bound infinite_var)+
        apply (rule SA_All(6))
     apply (subst extend_eqvt)
       apply (rule bij_swap supp_swap_bound infinite_var)+
     apply (rule arg_cong3[of _ _ _ _ _ _ extend])
    using SA_All(1,9) apply (metis bij_swap SA_All(4) Un_iff context_map_cong_id fun_upd_apply id_apply infinite_var supp_swap_bound wf_FFVars wf_context)
      apply simp
    using fresh SA_All(3) apply simp
    using x SA_All(8) unfolding same using Forall_inject_same by simp
qed (auto simp: Top)

lemma SA_AllE2[consumes 2, case_names SA_Trans_TVar SA_All]:
  assumes "Γ ⊢ S <: ∀X<:T⇩1. T⇩2" "X ∉ dom Γ"
    and TVr: "⋀Γ x U. ⟦ x<:U ∈ Γ ; Γ ⊢ U <: ∀ X <: T⇩1 . T⇩2 ; R Γ U (∀X<:T⇩1. T⇩2) ⟧ ⟹ R Γ (TVr x) (∀ X <: T⇩1 . T⇩2)"
    and Forall: "⋀Γ S⇩1 S⇩2. ⟦ Γ ⊢ T⇩1 <: S⇩1 ; Γ,, X<:T⇩1 ⊢ S⇩2 <: T⇩2 ⟧ ⟹ R Γ (∀X<:S⇩1. S⇩2) (∀ X <: T⇩1 . T⇩2)"
  shows "R Γ S (∀X<:T⇩1. T⇩2)"
using assms(1,2) proof (binder_induction Γ S "∀X<:T⇩1. T⇩2" avoiding: Γ S "∀X<:T⇩1. T⇩2" rule: ty.strong_induct)
  case (SA_All Γ R⇩1 S⇩1 Y S⇩2 R⇩2)
  have 1: "∀Y<:S⇩1. S⇩2 = ∀X<:S⇩1. rrename_sftypeP (id(Y:=X,X:=Y)) S⇩2"
    apply (rule Forall_swap)
    using SA_All(6,9) well_scoped(1) by fastforce
  have fresh: "X ∉ dom Γ" "Y ∉ dom Γ"
    using SA_All(9) apply blast
    by (metis SA_All(6) fst_conv wf_ConsE wf_context)
  have fresh2: "X ∉ FFVars_sftypeP T⇩1" "Y ∉ FFVars_sftypeP T⇩1"
     apply (metis SA_All(4,8) in_mono fresh(1) sftypeP_inject(3) well_scoped(1))
    by (metis SA_All(4,8) in_mono fresh(2) sftypeP_inject(3) well_scoped(1))
  have same: "R⇩1 = T⇩1" using SA_All(8) sftypeP_inject(3) by blast
  have x: "∀Y<:T⇩1 . R⇩2 = ∀X<:T⇩1. rrename_sftypeP (id(Y:=X,X:=Y)) R⇩2"
    apply (rule Forall_swap)
    by (metis SA_All(8) Un_iff assms(1,2) in_mono sftypeP.set(4) well_scoped(2))
  show ?case unfolding 1
    apply (rule Forall)
     apply (metis SA_All(4,8) sftypeP_inject(3))
    apply (rule iffD2[OF arg_cong3[OF _ refl, of _ _ _ _ ty], rotated -1])
      apply (rule ty.equiv)
       apply (rule bij_swap supp_swap_bound infinite_var)+
        apply (rule SA_All(6))
     apply (subst extend_eqvt)
       apply (rule bij_swap supp_swap_bound infinite_var)+
     apply (rule arg_cong3[of _ _ _ _ _ _ extend])
    using fresh apply (metis bij_swap SA_All(4) Un_iff context_map_cong_id fun_upd_apply id_apply infinite_var supp_swap_bound wf_FFVars wf_context)
      apply simp
    using fresh2 unfolding same apply (metis bij_swap fun_upd_apply id_apply infinite_var supp_swap_bound sftypeP.rrename_cong_ids)
    using SA_All(8) x Forall_inject_same unfolding same by simp
qed (auto simp: TVr)

lemma ty_transitivity : "⟦ Γ ⊢ S <: Q ; Γ ⊢ Q <: T ⟧ ⟹ Γ ⊢ S <: T"
  and ty_narrowing : "⟦ (Γ ,, X <: Q),, Δ ⊢ M <: N ; Γ ⊢ R <: Q ⟧ ⟹ (Γ,, X <: R),, Δ ⊢ M <: N"

      fix Γ S T
      show ty_trans: "Γ ⊢ S <: TVr Y ⟹ Γ ⊢ TVr Y <: T ⟹ Γ ⊢ S <: T"
        by (induction Γ S "TVr Y" rule: ty.induct) auto
    } note ty_trans = this
    {
      case 2
      then show ?case
      proof (binder_induction "Γ,, X <: TVr Y,, Δ" M N arbitrary: Δ avoiding: X "dom Γ" "dom Δ" rule: ty.strong_induct)
        case (SA_Trans_TVar Z U T Δ')
        show ?case
        proof (cases "X = Z")
          case True
          then have u: "U = TVr Y" using SA_Trans_TVar(1,2) context_determ wf_context by blast
          have "FFVars_sftypeP (TVr Y) ⊆ dom (Γ,, Z <: R,, Δ')" using SA_Trans_TVar(2) True u well_scoped(1) by fastforce
          then have "Γ,, Z <: R ,, Δ' ⊢ TVr Y <: T" using SA_Trans_TVar True u by auto
          moreover have "Γ,, Z <: R,, Δ' ⊢ R <: TVr Y" using ty_weakening[OF ty_weakening_extend[OF SA_Trans_TVar(4)]]
            by (metis SA_Trans_TVar(5) True wf_ConsE wf_concatD)

          then show ?thesis unfolding True u using ty.SA_Trans_TVar by auto
        next

          have x: "FFVars_sftypeP (U) ⊆ dom (Γ,, X <: R,, Δ')" using SA_Trans_TVar(2) well_scoped(1) by fastforce
          show ?thesis
            apply (rule ty.SA_Trans_TVar)
            using SA_Trans_TVar False x by auto
        qed
      next

        then show ?case by auto
      next

          apply (rule wf_Cons)
          using SA_All UnI1 image_iff by auto
        have "Γ,, X <: R,, (Δ',, Z <: T⇩1) ⊢ S⇩2 <: T⇩2"
          by (smt (verit, del_insts) "1" SA_All.hyps(6-8) append_Cons fst_conv image_Un image_insert list.simps(15) set_append well_scoped(1) well_scoped(2))

      next
        case (SA_Arrow T⇩1 S⇩1 S⇩2 T⇩2 Δ')
        then show ?case by auto
      next
        case (SA_All T⇩1 S⇩1 Z S⇩2 T⇩2 Δ')
        have 1: "wf (Γ,, Y <: R,, Δ',, Z <: T⇩1)"
          apply (rule wf_Cons)
          using SA_All UnI1 image_iff by auto
        have "Γ,, Y <: R,, (Δ',, Z <: T⇩1) ⊢ S⇩2 <: T⇩2"
          apply (rule SA_All(8))
                 apply (simp add: SA_All 1)+
          using SA_All.hyps(7) well_scoped(1) apply fastforce
             apply (simp add: SA_All 1)
          using SA_All.hyps(7) well_scoped(2) apply fastforce
            using SA_All by (auto simp: 1)
        then show ?case using SA_All by auto
      qed (rule context_set_bd_UNIV | blast)+
    }
  qed simp_all

  show "⟦ Γ ⊢ S <: Q ; Γ ⊢ Q <: T ⟧ ⟹ Γ ⊢ S <: T" using ty_trans by blast
  show "⟦(Γ,, X <: Q),, Δ ⊢ M <: N ; Γ ⊢ R <: Q ⟧ ⟹ (Γ,, X <: R),, Δ ⊢ M <: N"
    apply (rule ty_narrow)
        apply assumption+
    using narrow_wf well_scoped wf_context apply metis
    using well_scoped by fastforce+
qed

lemma ty_narrowing_aux:
  assumes ty_trans: "⋀Γ S T. ⟦ Γ ⊢ S <: Q ; Γ ⊢ Q <: T ⟧ ⟹ Γ ⊢ S <: T"
  and "(Γ,, X <: Q),, Δ ⊢ M <: N" "Γ ⊢ R <: Q"
shows "(Γ,, X <: R),, Δ ⊢ M <: N"
proof -
  have 1: "wf (Γ,, X <: R,, Δ)" using assms(2-3) narrow_wf well_scoped(1) wf_context by blast
  show ?thesis using assms(2-3) 1
  proof (induction "(Γ,, X <: Q),, Δ" M N arbitrary: Δ rule: ty.induct)
    case (SA_Trans_TVar Y U T Δ)
    show ?case
    proof (cases "X = Y")
      case True
      then have "Γ,, X <: R,, Δ ⊢ U <: T" using SA_Trans_TVar by auto
      then show ?thesis
        by (metis SA_Trans_TVar(1,2,4) True context_determ list.set_intros(1) ty.SA_Trans_TVar ty_trans ty_weakening ty_weakening_extend wf_ConsE wf_concatD wf_context)
    next
      case False
      then show ?thesis using SA_Trans_TVar by auto
    qed
  next
    case (SA_All T⇩1 S⇩1 x S⇩2 T⇩2 Δ)
    then show ?case by (metis Cons_eq_appendI narrow_wf ty.SA_All well_scoped(1) wf_context)
  qed (auto simp: ty.intros)
qed

theorem ty_transitivity2: "⟦ Γ ⊢ S <: Q ; Γ ⊢ Q <: T ⟧ ⟹ Γ ⊢ S <: T"
proof (binder_induction Q arbitrary: Γ S T avoiding: "dom Γ" rule: sftypeP.strong_induct)
  case (TVr x Γ S T)
  then show ?case
    by (induction Γ S "TVr x") auto
next
  case (Fun x1 x2 Γ S T)
  from Fun(4,3) show ?case
  proof (induction Γ "x1 → x2" T)
    case (SA_Top Γ)
    then show ?case using well_scoped by blast
  next
    case outer: (SA_Arrow Γ T⇩1 S⇩1 S⇩2 T⇩2)
    from outer(6,1,3,5) show ?case
    proof (induction Γ S "x1 → x2")
      case (SA_Arrow Γ U⇩1 R⇩1 R⇩2 U⇩2)
      show ?case
        apply (rule ty.SA_Arrow)
         apply (metis Fun(1) SA_Arrow(1,5,6,8) sftypeP_inject(2))
        by (metis Fun(2) SA_Arrow(3,5,7,8) sftypeP_inject(2))
    qed auto
  qed auto
next
  case (Forall X Q⇩1 Q⇩2 Γ S T)
  from Forall(4,1,5,1) show ?case
  proof (induction rule: SA_AllE2)
    case outer: (SA_All Γ S⇩1 S⇩2)
    from outer(3,4,1,2,4) show ?case
    proof (induction rule: SA_AllE1)
      case (SA_Trans_TVar Γ)
      then show ?case by (meson SA_All SA_Top well_scoped(1))
    next
      case (SA_All Γ T⇩1 T⇩2)
      then have "Γ,, X <: T⇩1 ⊢ S⇩2 <: Q⇩2"
        using ty_narrowing_aux[where Δ="[]", unfolded append.simps] Forall(2) by blast
      then have "Γ,, X <: T⇩1 ⊢ S⇩2 <: T⇩2" using Forall(3) SA_All(2) by blast
      moreover have "Γ ⊢ T⇩1 <: S⇩1" by (rule Forall(2)[OF SA_All(1,3)])
      ultimately show ?case by blast
    qed
  qed blast
qed auto


corollary ty_narrowing2: "⟦ Γ,, X <: Q,, Δ ⊢ M <: N ; Γ ⊢ R <: Q ⟧ ⟹ Γ,, X <: R,, Δ ⊢ M <: N"
  using ty_narrowing_aux ty_transitivity2 by blast
