  assumes "bij σ" "|supp σ| <o |UNIV::'a set|" shows "
 (⋀a'. a'∈TFV T2 - {x::'a::var} ⟹ σ a' = a') ⟹ All x T1 T2 = All (σ x) T1 (permute_type σ T2)"

  apply (unfold type.TT_inject0)
  apply (unfold set3_type_pre_def set2_type_pre_def comp_def Abs_type_pre_inverse[OF UNIV_I] map_sum.simps

  apply (rule exI[of _ σ])
  apply (rule conjI assms)+
   apply (unfold id_on_def atomize_all atomize_imp)[1]
   apply (rule impI)
   apply assumption
  apply (rule refl)
  done

  assumes "Γ ⊢ S <: T"
  shows "S closed_in Γ" "T closed_in Γ"

  case (SA_Trans_TVar x U Γ T)
  {
    case 1 then show ?case using SA_Trans_TVar
      by (metis fst_conv imageI singletonD subsetI type.set(1))
  next
    case 2 then show ?case using SA_Trans_TVar by simp
  }
next
  case (SA_TRec Γ Y X)
  {
    case 1
    then show ?case unfolding type.set
    proof safe
      fix x T
      assume a: "T ∈ values X" "x ∈ TFV T"
      from ‹T ∈ values X› obtain l where 1: "(l, T) ∈∈ X"
        by (meson values_lfin)
      then show "x ∈ dom Γ" using SA_TRec(3) a by fast
    qed
  next
    case 2
    then show ?case unfolding type.set
      by (auto dest!: values_lfin SA_TRec(4))
  }
qed auto

lemma permute_type_eq_substT_TVr:
assumes "bij (σ::'a⇒'a)" "|supp σ| <o |UNIV::'a::var set|"
shows "permute_type σ = substT (TVr o σ)"
proof
  fix T
  show "permute_type σ T = substT (TVr o σ) T"
  proof (binder_induction T avoiding: "IImsupp_type (TVr ∘ σ)" T rule: type.strong_induct)
    case Bound
    then show ?case using assms
      by (auto simp: IImsupp_type_def infinite_UNIV intro!: type.Un_bound type.UN_bound type.SSupp_comp_bound)
  next
    case (All X T1 T2)
    then show ?case
      by (subst type.subst)
        (auto simp: assms infinite_UNIV SSupp_type_TVr_comp IImsupp_type_TVr_comp
          type_inject id_on_def FVars_substT supp_inv_bound imsupp_def not_in_supp_alt
          intro!: exI[of _ id])
  qed (auto simp: assms infinite_UNIV SSupp_type_TVr_comp intro: lfset.map_cong)
qed

lemma substT_comp:
  fixes f g ::"'a::var ⇒ 'a type"
  assumes "|SSupp_type f| <o |UNIV::'a set|"
  assumes "|SSupp_type g| <o |UNIV::'a set|"
  shows "substT g (substT f T) = substT (substT g o f) T"
proof (binder_induction T avoiding: "IImsupp_type f" "IImsupp_type g" T rule: type.strong_induct)
  case (All X T U)
  then show ?case
    apply (subst type.subst; simp add: assms)
    apply (subst type.subst; (simp add: assms FVars_substT)?)
    apply (metis VVr_eq_TVr singletonD type.in_IImsupp type.set(1))
    apply (subst type.subst; (simp add: assms SSupp_type_substT_bound)?)
    using IImsupp_type_substT assms(2) by blast
qed (auto simp: assms SSupp_type_substT_bound IImsupp_type_bound lfset.map_comp intro: lfset.map_cong)

lemma substT_cong:
  fixes f g ::"'a::var ⇒ 'a type"
  assumes "|SSupp_type f| <o |UNIV::'a set|"
  assumes "|SSupp_type g| <o |UNIV::'a set|"
  shows "(∀x ∈ TFV T. f x = g x) ⟹ substT f T = substT g T"
proof (binder_induction T avoiding: "IImsupp_type f" "IImsupp_type g" T rule: type.strong_induct)
  case (All X T U)
  then show ?case
    apply (subst (1 2) type.subst; simp add: assms)
    by (metis (mono_tags, lifting) DiffI IImsupp_type_def SSupp_type_def Un_iff mem_Collect_eq singletonD)
qed (auto simp: assms IImsupp_type_bound intro: lfset.map_cong)
