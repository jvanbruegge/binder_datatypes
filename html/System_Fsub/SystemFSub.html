  show "permute_type  T = substT (TVr o ) T"
  proof (binder_induction T avoiding: "IImsupp_type (TVr  )" T rule: type.strong_induct)
    case Bound
    then show ?case using assms
      by (auto simp: IImsupp_type_def infinite_UNIV intro!: type.Un_bound type.UN_bound type.SSupp_comp_bound)
  next
    case (All X T1 T2)

lemma substT_comp:
  fixes f g ::"'a::var  'a type"
  assumes "|SSupp_type f| <o |UNIV::'a set|"
  assumes "|SSupp_type g| <o |UNIV::'a set|"
  shows "substT g (substT f T) = substT (substT g o f) T"
proof (binder_induction T avoiding: "IImsupp_type f" "IImsupp_type g" T rule: type.strong_induct)
  case (All X T U)
  then show ?case
    apply (subst type.subst; simp add: assms)
    apply (subst type.subst; (simp add: assms FVars_substT)?)
    apply (metis VVr_eq_TVr singletonD type.in_IImsupp type.set(1))
    apply (subst type.subst; (simp add: assms SSupp_type_substT_bound)?)
    using IImsupp_type_substT assms(2) by blast
qed (auto simp: assms SSupp_type_substT_bound IImsupp_type_bound lfset.map_comp intro: lfset.map_cong)

lemma substT_cong:
  fixes f g ::"'a::var  'a type"
  assumes "|SSupp_type f| <o |UNIV::'a set|"
  assumes "|SSupp_type g| <o |UNIV::'a set|"
  shows "(x  TFV T. f x = g x)  substT f T = substT g T"
proof (binder_induction T avoiding: "IImsupp_type f" "IImsupp_type g" T rule: type.strong_induct)
  case (All X T U)
  then show ?case
    apply (subst (1 2) type.subst; simp add: assms)
    by (metis (mono_tags, lifting) DiffI IImsupp_type_def SSupp_type_def Un_iff mem_Collect_eq singletonD)
qed (auto simp: assms IImsupp_type_bound intro: lfset.map_cong)
